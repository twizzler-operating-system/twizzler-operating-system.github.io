29// Gets the global dma pool for the HAL in a given access direction. If it doesn't exist, create it.
30impl TwzHalStatic {
31    fn new() -> Self {
32        Self {
33            host_to_device: DmaPool::new(
34                DmaPool::default_spec(),
35                Access::HostToDevice,
36                DmaOptions::empty(),
37            ),
38            device_to_host: DmaPool::new(
39                DmaPool::default_spec(),
40                Access::DeviceToHost,
41                DmaOptions::empty(),
42            ),
43            bidirectional: DmaPool::new(
44                DmaPool::default_spec(),
45                Access::BiDirectional,
46                DmaOptions::empty(),
47            ),
48            available: Vec::new(),
49            shared: HashMap::new(),
50            dma_slices: HashMap::new(),
51        }
52    }
53
54    fn get_dma_pool(&self, dir: BufferDirection) -> &DmaPool {
55        match dir {
56            BufferDirection::DriverToDevice => &self.host_to_device,
57            BufferDirection::DeviceToDriver => &self.device_to_host,
58            BufferDirection::Both => &self.bidirectional,
59        }
60    }
61}

63unsafe impl Hal for TwzHal {
64    // Required methods
65    fn dma_alloc(pages: usize, direction: BufferDirection) -> (PhysAddr, NonNull<u8>) {
66        assert!(pages == 1, "Only 1 page supported");
67
68        let mut twzhal = get_twz_hal().lock().unwrap();
69        let mut dma_slice = if let Some(buffer) = twzhal.available.pop() {
70            buffer
71        } else {
72            let pool = twzhal.get_dma_pool(direction);
73            pool.allocate_array(pages, 0u8).unwrap()
74        };
75
76        let pin = dma_slice.pin().unwrap();
77        let phys_addr: virtio_drivers::PhysAddr =
78            u64::from(pin.into_iter().next().unwrap().addr()) as virtio_drivers::PhysAddr;
79        let virt = unsafe { NonNull::<u8>::new(dma_slice.get_mut().as_mut_ptr()) }.unwrap();
80        twzhal.shared.insert(phys_addr, dma_slice);
81        (phys_addr as PhysAddr, virt)
82    }
83
84    unsafe fn dma_dealloc(paddr: PhysAddr, _vaddr: NonNull<u8>, _pages: usize) -> i32 {
85        //tracing::info!("DEALLOC: {:?} {:p}", paddr, _vaddr);
86        let mut twzhal = get_twz_hal().lock().unwrap();
87        if let Some(dma_slice) = twzhal.shared.remove(&paddr) {
88            twzhal.available.push(dma_slice);
89        }
90        0
91    }

97    unsafe fn share(buffer: NonNull<[u8]>, _direction: BufferDirection) -> PhysAddr {
98        let buf_len = buffer.len();
99        //tracing::info!("SHARE: {:p} {} {:?}", buffer, buf_len, direction);
100        assert!(buf_len <= DMA_PAGE_SIZE, "Hal::Share(): Buffer too large");
101
102        let addr: usize = buffer.addr().into();
103        let page_align_addr = addr & !(DMA_PAGE_SIZE - 1);
104        let page_offset = addr - page_align_addr;
105
106        let mut twzhal = TWZHAL.get().unwrap().lock().unwrap();
107        let entry = twzhal.dma_slices.entry(page_align_addr).or_insert_with(|| {
108            tracing::debug!("setting up DMA for object, ptr = {:p}", buffer);
109            let handle =
110                twizzler_rt_abi::object::twz_rt_get_object_handle(buffer.as_ptr().cast::<u8>())
111                    .unwrap();
112            let offset = page_align_addr - handle.start().addr();
113            let obj = DmaObject::new(Object::<()>::from_handle(handle).unwrap());
114            obj.slice_region(
115                offset,
116                DMA_PAGE_SIZE,
117                Access::BiDirectional,
118                DmaOptions::empty(),
119            )
120        });
121
122        let pin = entry.pin().unwrap();
123        let phys = pin.backing[0].addr();
124
125        (phys.0 + page_offset as u64) as PhysAddr
126    }
