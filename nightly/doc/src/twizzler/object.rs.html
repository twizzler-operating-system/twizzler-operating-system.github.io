19/// Operations common to structured objects.
20pub trait TypedObject {
21    /// The base type of this object.
22    type Base: BaseType;
23
24    /// Returns a resolved reference to the object's base.
25    fn base_ref(&self) -> Ref<'_, Self::Base>;
26
27    fn base(&self) -> &Self::Base;
28}

30/// Operations common to all objects, with raw pointers.
31pub trait RawObject {
32    /// Get the underlying runtime handle for this object.
33    fn handle(&self) -> &ObjectHandle;
34
35    /// Get the object ID.
36    fn id(&self) -> ObjID {
37        self.handle().id()
38    }
39
40    /// Get a const pointer to the object base.
41    fn base_ptr<T>(&self) -> *const T {
42        self.lea(NULLPAGE_SIZE, size_of::<T>()).unwrap().cast()
43    }
44
45    /// Get a mut pointer to the object base.
46    fn base_mut_ptr<T>(&self) -> *mut T {
47        self.lea_mut(NULLPAGE_SIZE, size_of::<T>()).unwrap().cast()
48    }
49
50    /// Get a const pointer to the object metadata.
51    fn meta_ptr(&self) -> *const MetaInfo {
52        self.handle().meta().cast()
53    }
54
55    /// Get a mut pointer to the object metadata.
56    fn meta_mut_ptr(&self) -> *mut MetaInfo {
57        self.handle().meta().cast()
58    }
59
60    /// Get a const pointer to a given FOT entry.
61    fn fote_ptr(&self, idx: usize) -> Option<*const FotEntry> {
62        let offset: isize = (1 + idx).try_into().ok()?;
63        unsafe { Some((self.meta_ptr() as *const FotEntry).offset(-offset)) }
64    }
65
66    /// Get a mut pointer to a given FOT entry.
67    fn fote_ptr_mut(&self, idx: usize) -> Option<*mut FotEntry> {
68        let offset: isize = (1 + idx).try_into().ok()?;
69        unsafe { Some((self.meta_mut_ptr() as *mut FotEntry).offset(-offset)) }
70    }
71
72    /// Get a const pointer to given range of the object.
73    fn lea(&self, offset: usize, _len: usize) -> Option<*const u8> {
74        Some(unsafe { self.handle().start().add(offset) as *const u8 })
75    }
76
77    /// Get a mut pointer to given range of the object.
78    fn lea_mut(&self, offset: usize, _len: usize) -> Option<*mut u8> {
79        Some(unsafe { self.handle().start().add(offset) as *mut u8 })
80    }
81
82    /// If the pointer is local to this object, return the offset into the object. Otherwise, return
83    /// None.
84    fn ptr_local(&self, ptr: *const u8) -> Option<usize> {
85        if ptr.addr() >= self.handle().start().addr()
86            && ptr.addr() < self.handle().start().addr() + MAX_SIZE
87        {
88            Some(ptr.addr() - self.handle().start().addr())
89        } else {
90            None
91        }
92    }
93}

95impl RawObject for ObjectHandle {
96    fn handle(&self) -> &ObjectHandle {
97        self
98    }
99}

101pub struct Object<Base> {
102    handle: ObjectHandle,
103    _pd: PhantomData<*const Base>,
104}
105
106unsafe impl<Base> Sync for Object<Base> {}
107unsafe impl<Base> Send for Object<Base> {}
108
109impl<B> Clone for Object<B> {
110    fn clone(&self) -> Self {
111        Self {
112            handle: self.handle.clone(),
113            _pd: PhantomData,
114        }
115    }
116}
117
118impl<Base> Object<Base> {
119    pub fn tx(self) -> crate::tx::Result<TxObject<Base>> {
120        TxObject::new(self)
121    }
122
123    pub unsafe fn from_handle_unchecked(handle: ObjectHandle) -> Self {
124        Self {
125            handle,
126            _pd: PhantomData,
127        }
128    }
129
130    pub fn from_handle(handle: ObjectHandle) -> Result<Self, TwzError> {
131        // TODO: check base fingerprint
132        unsafe { Ok(Self::from_handle_unchecked(handle)) }
133    }
134
135    pub fn into_handle(self) -> ObjectHandle {
136        self.handle
137    }
138
139    pub unsafe fn cast<U>(self) -> Object<U> {
140        Object {
141            handle: self.handle,
142            _pd: PhantomData,
143        }
144    }
145
146    pub fn map(id: ObjID, flags: MapFlags) -> Result<Self, TwzError> {
147        // TODO: check base fingerprint
148        let handle = twizzler_rt_abi::object::twz_rt_map_object(id, flags)?;
149        tracing::debug!("map: {} {:?} => {:?}", id, flags, handle.start());
150        Self::from_handle(handle)
151    }
152
153    pub fn update(self) -> crate::tx::Result<Self> {
154        let id = self.id();
155        let flags = self.handle().map_flags();
156        drop(self);
157
158        Self::map(id, flags)
159    }
160
161    pub unsafe fn map_unchecked(id: ObjID, flags: MapFlags) -> Result<Self, TwzError> {
162        let handle = twizzler_rt_abi::object::twz_rt_map_object(id, flags)?;
163        unsafe { Ok(Self::from_handle_unchecked(handle)) }
164    }
165
166    pub fn id(&self) -> ObjID {
167        self.handle.id()
168    }
169}
170
171impl<Base> RawObject for Object<Base> {
172    fn handle(&self) -> &ObjectHandle {
173        &self.handle
174    }
175}

177impl<Base: BaseType> TypedObject for Object<Base> {
178    type Base = Base;
179
180    fn base_ref(&self) -> Ref<'_, Self::Base> {
181        let base = self.base_ptr();
182        unsafe { Ref::from_raw_parts(base, self.handle()) }
183    }
184
185    #[inline]
186    fn base(&self) -> &Self::Base {
187        unsafe { self.base_ptr::<Self::Base>().as_ref().unwrap_unchecked() }
188    }
