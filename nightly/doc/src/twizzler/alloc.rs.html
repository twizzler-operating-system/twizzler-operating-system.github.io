11pub trait Allocator: Clone {
12    fn alloc(&self, layout: Layout) -> Result<GlobalPtr<u8>, AllocError>;
13    unsafe fn dealloc(&self, ptr: GlobalPtr<u8>, layout: Layout);
14
15    fn alloc_tx(&self, layout: Layout, _tx: &impl TxHandle) -> crate::tx::Result<GlobalPtr<u8>> {
16        self.alloc(layout).map_err(|e| e.into())
17    }
18    unsafe fn dealloc_tx(
19        &self,
20        ptr: GlobalPtr<u8>,
21        layout: Layout,
22        _tx: &impl TxHandle,
23    ) -> crate::tx::Result<()> {
24        self.dealloc(ptr, layout);
25        Ok(())
26    }
27
28    fn realloc(
29        &self,
30        ptr: GlobalPtr<u8>,
31        layout: Layout,
32        newsize: usize,
33    ) -> Result<GlobalPtr<u8>, AllocError> {
34        let new_layout =
35            Layout::from_size_align(newsize, layout.align()).map_err(|_| AllocError)?;
36
37        let new_alloc = self.alloc(new_layout)?;
38        unsafe {
39            if !ptr.is_null() {
40                let new_res = new_alloc.resolve().mutable();
41                let old_res = ptr.resolve();
42                let copy_len = std::cmp::min(layout.size(), new_layout.size());
43                new_res.raw().copy_from(old_res.raw(), copy_len);
44            }
45        }
46        Ok(new_alloc)
47    }
48    fn realloc_tx(
49        &self,
50        ptr: GlobalPtr<u8>,
51        layout: Layout,
52        newsize: usize,
53        _tx: &impl TxHandle,
54    ) -> Result<GlobalPtr<u8>, AllocError> {
55        self.realloc(ptr, layout, newsize)
56    }
57}
