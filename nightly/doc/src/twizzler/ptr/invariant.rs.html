1use std::marker::PhantomData;
2
3use twizzler_abi::object::MAX_SIZE;
4use twizzler_rt_abi::object::ObjectHandle;
5
6use super::{GlobalPtr, Ref};
7use crate::{
8    marker::{Invariant, PhantomStoreEffect},
9    object::RawObject,
10    tx::TxObject,
11};
12
13#[repr(C)]
14#[derive(Debug, PartialEq, PartialOrd, Ord, Eq)]
15pub struct InvPtr<T: Invariant> {
16    value: u64,
17    _pse: PhantomStoreEffect,
18    _pd: PhantomData<*const T>,
19}
20
21impl<T: Invariant> InvPtr<T> {
22    fn get_this(this: *const Self) -> ObjectHandle {
23        twizzler_rt_abi::object::twz_rt_get_object_handle(this.cast()).unwrap()
24    }
25
26    pub fn global(&self) -> GlobalPtr<T> {
27        let fote = self.fot_index();
28        let obj = Self::get_this(self);
29        if fote == 0 {
30            return GlobalPtr::new(obj.id(), self.offset());
31        }
32        let re = twizzler_rt_abi::object::twz_rt_resolve_fot(&obj, fote, MAX_SIZE).unwrap();
33        GlobalPtr::new(re.id(), self.offset())
34    }
35
36    #[inline(always)]
37    fn local_resolve(&self) -> *const T {
38        let this = self as *const Self;
39        this.map_addr(|addr| (addr & !(MAX_SIZE - 1)) + self.offset() as usize)
40            .cast()
41    }
42
43    #[inline]
44    pub unsafe fn resolve(&self) -> Ref<'_, T> {
45        if core::intrinsics::likely(self.is_local()) {
46            Ref::from_ptr(self.local_resolve())
47        } else {
48            self.slow_resolve()
49        }
50    }
51
52    #[inline(never)]
53    unsafe fn slow_resolve(&self) -> Ref<'_, T> {
54        let fote = self.fot_index();
55        let obj = Self::get_this(self);
56        let re = twizzler_rt_abi::object::twz_rt_resolve_fot(&obj, fote, MAX_SIZE).unwrap();
57        let ptr = re
58            .lea(self.offset() as usize, size_of::<T>())
59            .unwrap()
60            .cast();
61        Ref::from_handle(re, ptr)
62    }
63
64    pub fn null() -> Self {
65        Self::from_raw_parts(0, 0)
66    }
67
68    pub fn from_raw_parts(idx: u32, offset: u64) -> Self {
69        Self {
70            value: ((idx as u64) << 48) | offset,
71            _pse: PhantomStoreEffect,
72            _pd: PhantomData,
73        }
74    }
75
76    #[inline(always)]
77    pub fn fot_index(&self) -> u64 {
78        self.value >> 48
79    }
80
81    #[inline(always)]
82    pub fn is_local(&self) -> bool {
83        self.fot_index() == 0
84    }
85
86    #[inline(always)]
87    pub fn offset(&self) -> u64 {
88        self.value & ((1 << 48) - 1)
89    }
90
91    #[inline]
92    pub fn raw(&self) -> u64 {
93        self.value
94    }
95
96    pub fn new<B>(tx: &TxObject<B>, gp: impl Into<GlobalPtr<T>>) -> crate::tx::Result<Self> {
97        let gp = gp.into();
98        if gp.id() == tx.id() {
99            return Ok(Self::from_raw_parts(0, gp.offset()));
100        }
101        let fote = tx.insert_fot(&gp.into())?;
102        Ok(Self::from_raw_parts(fote, gp.offset()))
103    }
104}
