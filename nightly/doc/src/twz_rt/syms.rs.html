4// This macro checks that our definition of a function is the same as that
5// defined by the bindings generated from bindgen. Thus the whole ABI
6// is type-checked! The only trick is that you have to specify the number of arguments.
7macro_rules! check_ffi_type {
8    ($f1:ident) => {
9        paste::paste! {
10            #[allow(dead_code, unused_variables, unused_assignments)]
11            fn [<__tc_ffi_ $f1>]() {
12                let mut x: unsafe extern "C-unwind" fn() -> _ = $f1;
13                x = twizzler_rt_abi::bindings::$f1;
14            }
15        }
16    };
17    ($f1:ident, _) => {
18        paste::paste! {
19            #[allow(dead_code, unused_variables, unused_assignments)]
20            fn [<__tc_ffi_ $f1>]() {
21                let mut x: unsafe extern "C-unwind" fn(_) -> _ = $f1;
22                x = twizzler_rt_abi::bindings::$f1;
23            }
24        }
25    };
26    ($f1:ident, _, _) => {
27        paste::paste! {
28            #[allow(dead_code, unused_variables, unused_assignments)]
29            fn [<__tc_ffi_ $f1>]() {
30                let mut x: unsafe extern "C-unwind" fn(_, _) -> _ = $f1;
31                x = twizzler_rt_abi::bindings::$f1;
32            }
33        }
34    };
35    ($f1:ident, _, _, _) => {
36        paste::paste! {
37            #[allow(dead_code, unused_variables, unused_assignments)]
38            fn [<__tc_ffi_ $f1>]() {
39                let mut x: unsafe extern "C-unwind" fn(_, _, _) -> _ = $f1;
40                x = twizzler_rt_abi::bindings::$f1;
41            }
42        }
43    };
44    ($f1:ident, _, _, _, _) => {
45        paste::paste! {
46            #[allow(dead_code, unused_variables, unused_assignments)]
47            fn [<__tc_ffi_ $f1>]() {
48                let mut x: unsafe extern "C-unwind" fn(_, _, _, _) -> _ = $f1;
49                x = twizzler_rt_abi::bindings::$f1;
50            }
51        }
52    };
53    ($f1:ident, _, _, _, _, _) => {
54        paste::paste! {
55            #[allow(dead_code, unused_variables, unused_assignments)]
56            fn [<__tc_ffi_ $f1>]() {
57                let mut x: unsafe extern "C-unwind" fn(_, _, _, _, _) -> _ = $f1;
58                x = twizzler_rt_abi::bindings::$f1;
59            }
60        }
61    };
62    ($f1:ident, _, _, _, _, _, _) => {
63        paste::paste! {
64            #[allow(dead_code, unused_variables, unused_assignments)]
65            fn [<__tc_ffi_ $f1>]() {
66                let mut x: unsafe extern "C-unwind" fn(_, _, _, _, _, _) -> _ = $f1;
67                x = twizzler_rt_abi::bindings::$f1;
68            }
69        }
70    };
71}

147pub unsafe extern "C-unwind" fn twz_rt_malloc(
148    sz: usize,
149    align: usize,
150    flags: alloc_flags,
151) -> *mut ::core::ffi::c_void {
152    let Ok(layout) = core::alloc::Layout::from_size_align(sz, align) else {
153        return core::ptr::null_mut();
154    };
155    if flags & ZERO_MEMORY != 0 {
156        OUR_RUNTIME.default_allocator().alloc_zeroed(layout).cast()
157    } else {
158        OUR_RUNTIME.default_allocator().alloc(layout).cast()
159    }
160}
161check_ffi_type!(twz_rt_malloc, _, _, _);

163#[no_mangle]
164pub unsafe extern "C-unwind" fn twz_rt_dealloc(
165    ptr: *mut ::core::ffi::c_void,
166    sz: usize,
167    align: usize,
168    flags: twizzler_rt_abi::bindings::alloc_flags,
169) {
170    let Ok(layout) = core::alloc::Layout::from_size_align(sz, align) else {
171        return;
172    };
173    if flags & ZERO_MEMORY != 0 {
174        let slice = unsafe { core::slice::from_raw_parts_mut(ptr.cast::<u8>(), sz) };
175        slice.fill(0);
176        core::hint::black_box(slice);
177    }
178    OUR_RUNTIME.default_allocator().dealloc(ptr.cast(), layout);
179}
180check_ffi_type!(twz_rt_dealloc, _, _, _, _);

307pub unsafe extern "C-unwind" fn twz_rt_fd_open(info: open_info) -> open_result {
308    let name = unsafe { core::slice::from_raw_parts(info.name.cast(), info.len) };
309    let name = core::str::from_utf8(name)
310        .map_err(|_| twizzler_rt_abi::error::ArgumentError::InvalidArgument);
311    match name {
312        Ok(name) => OUR_RUNTIME
313            .open(name, info.create.into(), info.flags.into())
314            .into(),
315        Err(e) => open_result {
316            err: TwzError::from(e).raw(),
317            fd: 0,
318        },
319    }

503fn twz_sf_to_std_sf(sf: SeekFrom) -> std::io::SeekFrom {
504    match sf {
505        SeekFrom::Start(pos) => std::io::SeekFrom::Start(pos),
506        SeekFrom::End(pos) => std::io::SeekFrom::End(pos),
507        SeekFrom::Current(pos) => std::io::SeekFrom::Current(pos),
508    }
509}

511#[no_mangle]
512pub unsafe extern "C-unwind" fn twz_rt_fd_seek(
513    fd: descriptor,
514    whence: whence,
515    offset: i64,
516) -> io_result {
517    let seek = match whence {
518        twizzler_rt_abi::bindings::WHENCE_START => SeekFrom::Start(offset as u64),
519        twizzler_rt_abi::bindings::WHENCE_END => SeekFrom::End(offset),
520        twizzler_rt_abi::bindings::WHENCE_CURRENT => SeekFrom::Current(offset),
521        _ => {
522            return io_result {
523                val: 0,
524                err: TwzError::INVALID_ARGUMENT.raw(),
525            }
526        }
527    };
528    OUR_RUNTIME.seek(fd, twz_sf_to_std_sf(seek)).into()
529}
530check_ffi_type!(twz_rt_fd_seek, _, _, _);

558fn result_id_to_bindings(value: Result<ObjID, TwzError>) -> objid_result {
559    match value {
560        Ok(id) => objid_result {
561            err: RawTwzError::success().raw(),
562            __bindgen_padding_0: 0,
563            val: id.raw(),
564        },
565        Err(err) => objid_result {
566            err: err.raw(),
567            __bindgen_padding_0: 0,
568            val: 0,
569        },
570    }

643    res_2: *mut map_result,
644) {

646        match OUR_RUNTIME.map_two_objects(
647            id_1.into(),
648            MapFlags::from_bits_truncate(flags_1),

650            MapFlags::from_bits_truncate(flags_2),

655            }
656            Err(e) => {

658                res_2.write(Err(e).into());
659            }
660        }
661    }
662}
