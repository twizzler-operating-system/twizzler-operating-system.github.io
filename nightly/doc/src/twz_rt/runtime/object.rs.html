31pub(crate) struct RuntimeHandleInfo {
32    refs: AtomicU64,
33    fot_cache: FotCache,
34    is_deleted: AtomicBool,
35}

37pub(crate) fn new_runtime_info() -> *mut RuntimeHandleInfo {
38    let rhi = Box::new(RuntimeHandleInfo {
39        refs: AtomicU64::new(1),
40        fot_cache: FotCache::new(),
41        is_deleted: AtomicBool::new(false),
42    });
43    Box::into_raw(rhi)
44}
45
46pub(crate) fn free_runtime_info(ptr: *mut RuntimeHandleInfo) {
47    if ptr.is_null() {
48        return;
49    }
50    let _boxed = unsafe { Box::from_raw(ptr) };
51}

53pub(crate) fn new_object_handle(id: ObjID, slot: usize, flags: MapFlags) -> ObjectHandle {
54    unsafe {
55        ObjectHandle::new(
56            id,
57            new_runtime_info().cast(),
58            (slot * MAX_SIZE) as *mut _,
59            (slot * MAX_SIZE + MAX_SIZE - NULLPAGE_SIZE) as *mut _,
60            flags,
61            MAX_SIZE - NULLPAGE_SIZE * 2,
62        )
63    }
64}

104    pub fn object_cmd(
105        &self,
106        handle: *mut object_handle,
107        cmd: object_cmd,
108        arg: *mut c_void,
109    ) -> Result<()> {
110        let cmd: ObjectCmd = cmd.try_into()?;
111        let handle = unsafe { &*handle };
112        match cmd {
113            ObjectCmd::Delete => {
114                sys_object_ctrl(
115                    handle.id.into(),
116                    ObjectControlCmd::Delete(DeleteFlags::empty()),
117                )?;
118                unsafe { &*handle.runtime_info.cast::<RuntimeHandleInfo>() }
119                    .is_deleted
120                    .store(true, Ordering::Release);
121                Ok(())
122            }
123            ObjectCmd::Sync => sys_map_ctrl(
124                handle.start.cast(),
125                MAX_SIZE,
126                MapControlCmd::Sync(arg.cast()),
127                0,
128            ),
129            ObjectCmd::Update => {
130                sys_map_ctrl(handle.start.cast(), MAX_SIZE, MapControlCmd::Update, 0)
131            }
132        }
133    }

153    pub fn get_object_handle_from_ptr(&self, ptr: *const u8) -> Result<object_handle> {
154        if let Some(handle) = self.object_manager.lock().get_handle(ptr) {
155            return Ok(handle);
156        }
157
158        let slot = ptr as usize / MAX_SIZE;
159        let Some(id) = self.get_id_from_heap_ptr(ptr) else {
160            let map = sys_object_read_map(None, slot)?;
161            return Ok(object_handle {
162                id: map.id.raw(),
163                start: (slot * MAX_SIZE) as *mut c_void,
164                map_flags: map.flags.bits(),
165                ..Default::default()
166            });
167        };
168        Ok(object_handle {
169            id: id.raw(),
170            start: (slot * MAX_SIZE) as *mut c_void,
171            map_flags: (MapFlags::READ | MapFlags::WRITE).bits(),
172            ..Default::default()
173        })
174    }

176    pub fn insert_fot(&self, handle: *mut object_handle, fot: *const u8) -> Result<u32> {
177        //tracing::warn!("TODO: insert FOT entry");
178        let handle = unsafe { &*handle };
179        // TODO: track max FOT entry
180        let _meta = unsafe { &*handle.meta };
181        let new_fot = unsafe { fot.cast::<FotEntry>().read() };
182        for i in 1..u32::MAX {
183            let ptr = unsafe { &mut *handle.meta.cast::<FotEntry>().sub((i + 1) as usize) };
184            let flags = FotFlags::from_bits_truncate(ptr.flags.load(Ordering::SeqCst));
185
186            if flags.contains(FotFlags::ALLOCATED)
187                && flags.contains(FotFlags::ACTIVE)
188                && !flags.contains(FotFlags::DELETED)
189            {
190                if ptr.values == new_fot.values && ptr.resolver == new_fot.resolver {
191                    return Ok(i);
192                }
193            }
194
195            if flags.contains(FotFlags::DELETED)
196                || (!flags.contains(FotFlags::ACTIVE) && !flags.contains(FotFlags::ALLOCATED))
197            {
198                if let Ok(_) = ptr.flags.compare_exchange(
199                    flags.bits(),
200                    FotFlags::ALLOCATED.bits(),
201                    Ordering::SeqCst,
202                    Ordering::SeqCst,
203                ) {
204                    let mut flags =
205                        FotFlags::from_bits_truncate(new_fot.flags.load(Ordering::SeqCst));
206                    flags.set(FotFlags::DELETED, false);
207                    flags.set(FotFlags::ALLOCATED, true);
208                    flags.set(FotFlags::ACTIVE, true);
209                    ptr.values = new_fot.values;
210                    ptr.resolver = new_fot.resolver;
211                    ptr.flags.store(flags.bits(), Ordering::SeqCst);
212                    return Ok(i);
213                }
214            }
215        }
216        Err(ResourceError::OutOfResources.into())
217    }

219    fn read_fot_entry(&self, handle: &object_handle, idx: u64) -> Result<FotEntry> {
220        let ptr = unsafe { &*handle.meta.cast::<FotEntry>().sub((idx + 1) as usize) };
221        let flags = FotFlags::from_bits_truncate(ptr.flags.load(Ordering::SeqCst));
222        if flags.contains(FotFlags::DELETED)
223            || !flags.contains(FotFlags::ACTIVE)
224            || !flags.contains(FotFlags::ALLOCATED)
225        {
226            return Err(ObjectError::InvalidFote.into());
227        }
228        if flags.contains(FotFlags::RESOLVER) {
229            return Err(TwzError::NOT_SUPPORTED);
230        }
231        let val = unsafe { (ptr as *const FotEntry).read_volatile() };
232
233        let flags = FotFlags::from_bits_truncate(val.flags.load(Ordering::SeqCst));
234        if flags.contains(FotFlags::DELETED)
235            || !flags.contains(FotFlags::ACTIVE)
236            || !flags.contains(FotFlags::ALLOCATED)
237        {
238            return Err(ObjectError::InvalidFote.into());
239        }
240        Ok(val)
241    }

243    pub fn resolve_fot(
244        &self,
245        handle: *mut object_handle,
246        idx: u64,
247        _valid_len: usize,
248        map_flags: MapFlags,
249    ) -> Result<ObjectHandle> {
250        if idx == 0 || handle.is_null() {
251            return Err(TwzError::INVALID_ARGUMENT);
252        }
253        let handle = unsafe { &*handle };
254        tracing::trace!("Resolving FOT: {:x}", handle.id);
255        let entry = self.read_fot_entry(handle, idx)?;
256        let id = ObjID::from_parts(entry.values);
257
258        let res_handle = self.map_object(id, map_flags)?;
259        unsafe { &*handle.runtime_info.cast::<RuntimeHandleInfo>() }
260            .fot_cache
261            .insert(idx, map_flags, res_handle.clone());
262        Ok(res_handle)
263    }

265    pub fn resolve_fot_local(
266        &self,
267        ptr: *mut u8,
268        idx: u64,
269        _valid_len: usize,
270        flags: MapFlags,
271    ) -> *mut u8 {
272        if let Some(handle) = self.object_manager.lock().get_handle(ptr) {
273            tracing::trace!("Resolving FOT local: {:x}", handle.id);
274            let rtinfo: *const RuntimeHandleInfo = handle.runtime_info.cast();
275            unsafe {
276                return (&*rtinfo)
277                    .fot_cache
278                    .resolve_cached_ptr(idx, flags)
279                    .unwrap_or(core::ptr::null_mut());
280            }
281        }
282        core::ptr::null_mut()
283    }

285    pub fn map_two_objects(
286        &self,
287        in_id_a: ObjID,
288        in_flags_a: MapFlags,
289        in_id_b: ObjID,
290        in_flags_b: MapFlags,
291    ) -> Result<(ObjectHandle, ObjectHandle)> {
292        let mapping =
293            monitor_api::monitor_rt_object_pair_map(in_id_a, in_flags_a, in_id_b, in_flags_b)?;
294
295        let handle = new_object_handle(in_id_a, mapping.0.slot, in_flags_a);
296        let handle2 = new_object_handle(in_id_b, mapping.1.slot, in_flags_b);
297        Ok((handle, handle2))
298    }

314/// Per-compartment object management.
315pub struct ObjectHandleManager {
316    cache: HandleCache,
317}
318
319impl ObjectHandleManager {
320    pub const fn new() -> Self {
321        Self {
322            cache: HandleCache::new(),
323        }
324    }
325
326    /// Map an object with this manager. Will call to monitor if needed.
327    pub fn map_object(&mut self, key: ObjectMapKey) -> Result<ObjectHandle> {
328        if let Some(handle) = self.cache.activate(key) {
329            let oh = ObjectHandle::from_raw(handle);
330            let oh2 = oh.clone();
331            std::mem::forget(oh);
332            return Ok(oh2);
333        }
334        let mapping = monitor_api::monitor_rt_object_map(key.0, key.1)?;
335        let handle = new_object_handle(key.0, mapping.slot, key.1).into_raw();
336        self.cache.insert(handle);
337        Ok(ObjectHandle::from_raw(handle))
338    }
339
340    /// Get an object handle from a pointer to within that object.
341    pub fn get_handle(&mut self, ptr: *const u8) -> Option<object_handle> {
342        let handle = self.cache.activate_from_ptr(ptr)?;
343        let oh = ObjectHandle::from_raw(handle);
344        let oh2 = oh.clone().into_raw();
345        std::mem::forget(oh);
346        Some(oh2)
347    }
348
349    /// Release a handle. If all handles have been released, calls to monitor to unmap.
350    pub fn release(&mut self, handle: *mut object_handle, mut flags: release_flags) {
351        let handle = unsafe { handle.as_mut().unwrap() };
352        if unsafe { &*handle.runtime_info.cast::<RuntimeHandleInfo>() }
353            .is_deleted
354            .load(Ordering::Acquire)
355        {
356            flags |= RELEASE_NO_CACHE;
357        }
358        self.cache.release(handle, flags);
359    }
