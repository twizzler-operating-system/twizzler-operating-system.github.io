26#[derive(Clone)]
27pub struct RawFile {
28    pub(crate) pos: Arc<AtomicU64>,
29    len: Arc<AtomicU64>,
30    handle: ObjectHandle,
31}
32
33impl RawFile {
34    fn update_len(&self) {
35        if let Some(me) = self.handle.find_meta_ext(MEXT_SIZED) {
36            self.len
37                .store(me.value.load(Ordering::SeqCst), Ordering::SeqCst);
38        }
39    }
40
41    pub fn open(obj_id: ObjID, flags: MapFlags) -> Result<Self> {
42        let handle = OUR_RUNTIME
43            .map_object(obj_id, flags | MapFlags::NO_NULLPAGE)
44            .unwrap();
45        let len = if let Some(me) = handle.find_meta_ext(MEXT_SIZED) {
46            me.value.load(Ordering::SeqCst)
47        } else {
48            if flags.contains(MapFlags::WRITE) {
49                unsafe { handle.set_meta_ext(MetaExt::new(MEXT_SIZED, 0))? };
50            }
51            0
52        };
53        Ok(Self {
54            pos: Arc::new(AtomicU64::new(0)),
55            len: Arc::new(AtomicU64::new(len)),
56            handle,
57        })
58    }
59
60    pub fn truncate(&self, new_len: u64) -> Result<()> {
61        if new_len > (MAX_SIZE - NULLPAGE_SIZE) as u64 {
62            return Err(ArgumentError::InvalidArgument.into());
63        }
64        self.len.store(new_len, Ordering::SeqCst);
65        let me = MetaExt::new(MEXT_SIZED, new_len);
66        unsafe { self.handle.set_meta_ext(me)? };
67        Ok(())
68    }
69}
70
71impl Fd for RawFile {
72    fn read(
73        &self,
74        buf: &mut [u8],
75        _flags: twizzler_rt_abi::io::IoFlags,
76        a_offset: Option<u64>,
77        _ep: Option<&mut twizzler_rt_abi::io::Endpoint>,
78    ) -> Result<usize> {
79        self.update_len();
80        let offset = a_offset.unwrap_or(self.pos.load(Ordering::SeqCst));
81        let len = self.len.load(Ordering::SeqCst);
82        if offset >= len {
83            return Ok(0);
84        }
85        let copy_len = buf.len().min((len - offset) as usize);
86        let data = unsafe {
87            core::slice::from_raw_parts(self.handle.start().add(offset as usize), copy_len)
88        };
89        buf[0..copy_len].copy_from_slice(data);
90        if a_offset.is_none() {
91            self.pos.store(offset + copy_len as u64, Ordering::SeqCst);
92        }
93        Ok(copy_len)
94    }
95
96    fn write(
97        &self,
98        buf: &[u8],
99        _flags: twizzler_rt_abi::io::IoFlags,
100        a_offset: Option<u64>,
101        _to: Option<&twizzler_rt_abi::io::Endpoint>,
102    ) -> Result<usize> {
103        let offset = a_offset.unwrap_or(self.pos.load(Ordering::SeqCst));
104        let write_len = buf.len();
105        let end_pos = offset + write_len as u64;
106        if end_pos > (MAX_SIZE - NULLPAGE_SIZE) as u64 {
107            return Err(TwzError::INVALID_ARGUMENT);
108        }
109        let len = self.len.load(Ordering::SeqCst);
110        if end_pos > len {
111            self.len.store(end_pos, Ordering::SeqCst);
112            let me = twizzler_rt_abi::object::MetaExt::new(MEXT_SIZED, end_pos);
113            unsafe { self.handle.set_meta_ext(me)? };
114        }
115        unsafe {
116            let dest = self.handle.start().add(offset as usize);
117            core::ptr::copy_nonoverlapping(buf.as_ptr(), dest, write_len);
118        }
119        if a_offset.is_none() {
120            self.pos.store(offset + write_len as u64, Ordering::SeqCst);
121        }
122        self.handle.cmd(ObjectCmd::Sync, null_mut::<()>())?;
123        Ok(write_len)
124    }
125
126    fn stat(&self) -> Result<FdInfo> {
127        self.update_len();
128        Ok(FdInfo {
129            kind: twizzler_rt_abi::fd::FdKind::Regular,
130            size: self.len.load(Ordering::SeqCst),
131            flags: twizzler_rt_abi::fd::FdFlags::empty(),
132            id: self.handle.id().raw(),
133            unix_mode: S_IFREG | S_IRWXG | S_IRWXO | S_IRWXU,
134            accessed: std::time::Duration::ZERO,
135            modified: std::time::Duration::ZERO,
136            created: std::time::Duration::ZERO,
137        })
138    }
139
140    fn seek(&self, pos: SeekFrom) -> Result<usize> {
141        self.update_len();
142        let new_pos: i64 = match pos {
143            SeekFrom::Start(x) => x as i64,
144            SeekFrom::End(x) => (self.len.load(Ordering::SeqCst) as i64) - x,
145            SeekFrom::Current(x) => (self.pos.load(Ordering::SeqCst) as i64) + x,
146        };
147
148        if new_pos < 0 {
149            Err(ArgumentError::InvalidArgument.into())
150        } else {
151            self.pos.store(new_pos as u64, Ordering::SeqCst);
152            Ok(new_pos as usize)
153        }
154    }
155
156    fn flush(&self) -> Result<()> {
157        Ok(())
158    }
159
160    fn fd_cmd(&self, cmd: u32, arg: *const u8, _ret: *mut u8) -> Result<()> {
161        match cmd {
162            twizzler_rt_abi::bindings::FD_CMD_TRUNCATE => {
163                let new_len = unsafe { *(arg as *const u64) };
164                self.truncate(new_len)?;
165                Ok(())
166            }
167            _ => Err(ArgumentError::InvalidArgument.into()),
168        }
169    }
170
171    fn get_config(&self, _reg: u32, _val: *mut std::ffi::c_void, _val_len: usize) -> Result<()> {
172        Err(ErrorKind::Unsupported.into())
173    }
174
175    fn set_config(&self, _reg: u32, _val: *const std::ffi::c_void, _val_len: usize) -> Result<()> {
176        Err(ErrorKind::Unsupported.into())
177    }
178
179    fn waitpoint(
180        &self,
181        _kind: twizzler_rt_abi::bindings::wait_kind,
182    ) -> Result<(ThreadSyncSleep, bool)> {
183        if let Some(me) = self.handle.find_meta_ext(MEXT_SIZED) {
184            let ready = self.pos.load(Ordering::SeqCst) < me.value.load(Ordering::SeqCst);
185            Ok(((&me.value, self.pos.load(Ordering::SeqCst)).into(), ready))
186        } else {
187            Err(ErrorKind::Unsupported.into())
188        }
189    }
190
191    fn shutdown(&self, _sh: std::net::Shutdown) -> Result<()> {
192        Ok(())
193    }
