29struct TraceSink {
30    _prime: ObjectHandle,
31    current: ObjectHandle,
32    pos: u64,
33    start: Instant,
34}
35
36const TRACE_START: u64 = NULLPAGE_SIZE as u64 * 2;
37fn write_base(handle: &ObjectHandle, start: u64) {
38    unsafe {
39        handle
40            .start()
41            .add(NULLPAGE_SIZE)
42            .cast::<TraceBase>()
43            .write(TraceBase {
44                end: AtomicU64::new(start),
45                start,
46            });
47    };
48}
49
50impl TraceSink {
51    pub fn new(id: ObjID) -> Option<Self> {
52        tracing::info!("initializing runtime trace object {}", id);
53        let prime = OUR_RUNTIME
54            .map_object(id, MapFlags::READ | MapFlags::WRITE)
55            .ok()?;
56        let start = TRACE_START;
57        write_base(&prime, start);
58        let this = Self {
59            current: prime.clone(),
60            _prime: prime,
61            pos: start,
62            start: Instant::now(),
63        };
64        this.write_endpoint(start);
65        Some(this)
66    }
67
68    fn write_endpoint(&self, new_end: u64) {
69        let end = unsafe { &*self.current.start().add(NULLPAGE_SIZE).cast::<AtomicU64>() };
70        end.store(new_end, Ordering::SeqCst);
71        let _ = sys_thread_sync(
72            &mut [ThreadSync::new_wake(ThreadSyncWake::new(
73                ThreadSyncReference::Virtual(end),
74                usize::MAX,
75            ))],
76            None,
77        )
78        .inspect_err(|e| tracing::warn!("failed to wake tracing end point: {}", e));
79    }
80
81    pub fn push_record<T: Copy + core::fmt::Debug>(
82        &mut self,
83        mut record: TraceEntryHead,
84        trace_data: Option<T>,
85    ) {
86        if trace_data.is_some() {
87            record.flags.insert(TraceEntryFlags::HAS_DATA);
88        }
89        let data = (&record) as *const TraceEntryHead as *const u8;
90        let len = size_of::<TraceEntryHead>();
91        unsafe {
92            let start = self.current.start().add(self.pos as usize);
93            let slice = core::slice::from_raw_parts_mut(start, len);
94            let data = core::slice::from_raw_parts(data, len);
95            slice.copy_from_slice(data);
96        }
97        self.pos += len as u64;
98
99        if let Some(trace_data) = trace_data {
100            let data_len = (size_of::<TraceData<()>>() + size_of::<T>())
101                .next_multiple_of(align_of::<TraceEntryHead>());
102            let data_header = TraceData {
103                resv: 0,
104                len: data_len as u32,
105                flags: 0,
106                data: (),
107            };
108            let data = (&data_header) as *const TraceData<()> as *const u8;
109            unsafe {
110                let start = self.current.start().add(self.pos as usize);
111                let slice = core::slice::from_raw_parts_mut(start, size_of::<TraceData<()>>());
112                let data = core::slice::from_raw_parts(data, size_of::<TraceData<()>>());
113                slice.copy_from_slice(data);
114            }
115            self.pos += size_of::<TraceData<()>>() as u64;
116
117            let data = (&trace_data) as *const T as *const u8;
118            unsafe {
119                let start = self.current.start().add(self.pos as usize);
120                let slice = core::slice::from_raw_parts_mut(start, size_of::<T>());
121                let data = core::slice::from_raw_parts(data, size_of::<T>());
122                slice.copy_from_slice(data);
123            }
124            self.pos += size_of::<T>() as u64;
125        }
126        self.pos = self
127            .pos
128            .next_multiple_of(align_of::<TraceEntryHead>() as u64);
129        self.write_endpoint(self.pos);
130    }
131
132    pub fn check_space(&mut self) -> bool {
133        const TRACE_MAX: u64 = MAX_SIZE as u64 / 2;
134        if self.pos >= TRACE_MAX {
135            let create = ObjectCreate::default();
136            let new_id = match sys_object_create(create, &[], &[]) {
137                Err(e) => {
138                    tracing::warn!("failed to allocate new runtime tracing object: {}", e);
139                    return false;
140                }
141                Ok(id) => id,
142            };
143            tracing::info!("initializing continued runtime trace object {}", new_id);
144            let new_handle = match OUR_RUNTIME.map_object(new_id, MapFlags::READ | MapFlags::WRITE)
145            {
146                Err(e) => {
147                    tracing::warn!("failed to map new runtime tracing object: {}", e);
148                    return false;
149                }
150                Ok(id) => id,
151            };
152
153            let new_start = TRACE_START;
154            write_base(&new_handle, new_start);
155
156            self.push_record::<()>(TraceEntryHead::new_next_object(new_id), None);
157
158            self.current = new_handle;
159            self.pos = new_start;
160            self.write_endpoint(new_start);
161        }
162        true
163    }
164}

166struct OnceTraceSink {
167    key: &'static str,
168    val: OnceLock<Option<Mutex<TraceSink>>>,
169}
170
171impl OnceTraceSink {
172    pub const fn new(key: &'static str) -> Self {
173        Self {
174            key,
175            val: OnceLock::new(),
176        }
177    }
178
179    pub fn get(&self) -> Option<&Mutex<TraceSink>> {
180        self.val
181            .get_or_init(|| {
182                std::env::var(self.key)
183                    .ok()
184                    .and_then(|s| u128::from_str_radix(&s, 16).ok())
185                    .and_then(|id| TraceSink::new(id.into()).map(|ts| Mutex::new(ts)))
186            })
187            .as_ref()
188    }
189}
190
191static ENV_TRACE_OBJECT: OnceTraceSink = OnceTraceSink::new("TWZRT_TRACE_OBJECT");
192
193static DISABLE_ALLOC: AtomicBool = AtomicBool::new(false);
194
195#[allow(dead_code)]
196pub fn trace_runtime_alloc(addr: usize, layout: Layout, duration: Duration, is_free: bool) {
197    if !OUR_RUNTIME.state().contains(RuntimeState::READY) {
198        return;
199    }
200    if DISABLE_ALLOC.swap(true, Ordering::SeqCst) {
201        return;
202    }
203    if let Some(ts) = ENV_TRACE_OBJECT.get() {
204        let mut ts = ts.lock();
205        if ts.check_space() {
206            let record = TraceEntryHead {
207                thread: get_thread_id(),
208                sctx: get_sctx_id(),
209                mctx: 0.into(),
210                cpuid: 0,
211                time: (Instant::now() - ts.start).into(),
212                event: RUNTIME_ALLOC,
213                kind: TraceKind::Runtime,
214                extra_or_next: 0.into(),
215                flags: TraceEntryFlags::empty(),
216            };
217            let data = RuntimeAllocationEvent {
218                duration: duration.into(),
219                layout,
220                addr: addr as u64,
221                is_free,
222            };
223            ts.push_record(record, Some(data));
224        }
225    }
226    DISABLE_ALLOC.store(false, Ordering::SeqCst);
227}
