37struct PtrToInfo(*mut c_void);
38unsafe impl Send for PtrToInfo {}
39unsafe impl Sync for PtrToInfo {}
40static MON_RTINFO: OnceLock<Option<PtrToInfo>> = OnceLock::new();

42fn search_envs_enabled(envp: *mut *mut c_char, name: &str) -> bool {
43    unsafe {
44        let mut env = envp;
45        while !env.is_null() && !(*env).is_null() {
46            let s = std::ffi::CStr::from_ptr(*env);
47            if s.to_str().unwrap_or("").starts_with(name) {
48                return true;
49            }
50            env = env.add(1);
51        }
52        false
53    }
54}

58    pub fn exit(&self, code: i32) -> ! {
59        if self.state().contains(RuntimeState::READY) {
60            let id = crate::runtime::thread::with_current_thread(|ct| ct.id());
61            if id == 1 {
62                OUR_RUNTIME.close_fds();
63            }
64            twizzler_abi::syscall::sys_thread_exit(code as u64);
65        } else {
66            preinit_println!("runtime exit before runtime ready: {}", code);
67            preinit_abort();
68        }
69    }

71    pub fn abort(&self) -> ! {
72        if self.state().contains(RuntimeState::READY) {
73            preinit_abort();
74        } else {
75            preinit_println!("runtime abort before runtime ready");
76            preinit_abort();
77        }
78    }

88    pub fn cgetenv(&self, name: &CStr) -> *const c_char {
89        // TODO: this approach is very simple, but it leaks if the environment changes a lot.
90        static ENVMAP: Mutex<BTreeMap<String, CString>> = Mutex::new(BTreeMap::new());
91        let Ok(name) = name.to_str() else {
92            return core::ptr::null();
93        };
94        let Ok(val) = std::env::var(name) else {
95            return core::ptr::null();
96        };
97        let mut envmap = ENVMAP.lock().unwrap();
98        envmap
99            .entry(val.to_string())
100            .or_insert_with(|| CString::new(val.to_string()).unwrap())
101            .as_ptr()
102    }

104    pub fn runtime_entry(
105        &self,
106        rtinfo: *const RuntimeInfo,
107        std_entry: unsafe extern "C-unwind" fn(BasicAux) -> BasicReturn,
108        main: usize,
109    ) {
110        if OUR_RUNTIME.state().contains(RuntimeState::READY) {
111            return;
112        }
113        let rtinfo = unsafe { rtinfo.as_ref().unwrap() };
114        match rtinfo.kind {
115            RUNTIME_INIT_MONITOR => {
116                let init_info = unsafe {
117                    rtinfo
118                        .init_info
119                        .monitor
120                        .cast::<RuntimeInitInfo>()
121                        .as_ref()
122                        .unwrap()
123                };
124                let _ = MON_RTINFO.set(Some(PtrToInfo(init_info as *const _ as *mut _)));
125                self.init_for_monitor(init_info);
126            }
127            RUNTIME_INIT_COMP => {
128                let init_info = unsafe {
129                    rtinfo
130                        .init_info
131                        .comp
132                        .cast::<CompartmentInitInfo>()
133                        .as_ref()
134                        .unwrap()
135                };
136                let _ = MON_RTINFO.set(None);
137                let mut entry_stack = Vec::new();
138                entry_stack.push(rtinfo.argc);
139                if !rtinfo.args.is_null() {
140                    for arg in unsafe { core::slice::from_raw_parts(rtinfo.args, rtinfo.argc) } {
141                        entry_stack.push(*arg as usize);
142                    }
143                }
144                entry_stack.push(0);
145                if !rtinfo.envp.is_null() {
146                    let mut envp = rtinfo.envp;
147                    while !envp.is_null() && !unsafe { (*envp).is_null() } {
148                        entry_stack.push(unsafe { *envp } as usize);
149                        envp = unsafe { envp.add(1) };
150                    }
151                }
152                entry_stack.push(0);
153                self.init_for_compartment(init_info, entry_stack.as_mut_ptr());
154                std::mem::forget(entry_stack);
155            }
156            x => {
157                preinit_println!("unsupported runtime kind: {}", x);
158                preinit_abort();
159            }
160        }
161
162        let mut null_env: [*mut c_char; 4] = [
163            b"RUST_BACKTRACE=1\0".as_ptr() as *mut c_char,
164            std::ptr::null_mut(),
165            std::ptr::null_mut(),
166            std::ptr::null_mut(),
167        ];
168        let env_ptr = if rtinfo.envp.is_null() {
169            (&mut null_env).as_mut_ptr()
170        } else {
171            rtinfo.envp
172        };
173
174        if !unsafe { __twz_enable_libc_trace.is_null() } {
175            let twz_enable_libc_trace =
176                unsafe { std::mem::transmute::<_, extern "C" fn()>(__twz_enable_libc_trace) };
177
178            if search_envs_enabled(env_ptr, "TWZ_LIBC_TRACE") {
179                twz_enable_libc_trace();
180            }
181        }
182
183        // Step 3: call into libstd to finish setting up the standard library and call main
184        let ba = BasicAux {
185            argc: rtinfo.argc,
186            args: rtinfo.args,
187            env: env_ptr,
188            entry: main,
189        };
190
191        let ret = unsafe { std_entry(ba) };
192        self.exit(ret.code);
193    }

195    pub fn pre_main_hook(&self) -> Option<ExitCode> {
196        use twizzler_rt_abi::fd::NameRoot;
197        // TODO: control this with env vars
198        tracing::subscriber::set_global_default(
199            tracing_subscriber::fmt()
200                .with_max_level(Level::INFO)
201                .finish(),
202        )
203        .unwrap();
204        if self.state().contains(RuntimeState::IS_MONITOR) {
205            self.init_slots();
206            None
207        } else {
208            unsafe { self.set_runtime_ready() };
209            OUR_RUNTIME.init_fds();
210
211            let mut nr = self.nameroots.lock();
212            //twizzler_abi::klog_println!("got: {:?}", std::env::var("TWZ_RT_INITIAL_DIR"));
213
214            let current_dir = std::env::var("TWZ_RT_INITIAL_DIR").unwrap_or("/".to_string());
215
216            // TODO
217            nr.insert(NameRoot::Home, PathBuf::from("/"));
218            nr.insert(NameRoot::Root, PathBuf::from("/"));
219            nr.insert(NameRoot::Exe, PathBuf::from("/"));
220            nr.insert(NameRoot::Temp, PathBuf::from("/tmp"));
221            nr.insert(NameRoot::Current, PathBuf::from("/"));
222            if let Some(namer) = crate::runtime::file::get_naming_handle() {
223                if namer.lock().unwrap().change_namespace(&current_dir).is_ok() {
224                    nr.insert(NameRoot::Current, Path::new(&current_dir).to_path_buf());
225                }
226            }
227
228            let ret = match monitor_api::monitor_rt_comp_ctrl(
229                monitor_api::MonitorCompControlCmd::RuntimeReady,
230            ) {
231                Ok(ret) => ret,
232                _ => self.abort(),
233            };
234            ret
235        }
236    }

259impl ReferenceRuntime {
260    fn init_for_monitor(&self, init_info: &RuntimeInitInfo) {
261        let upcall_target = UpcallTarget::new(
262            Some(
263                twizzler_rt_abi::arch::__twz_rt_upcall_entry
264                    as unsafe extern "C-unwind" fn(_, _) -> !,
265            ),
266            Some(twizzler_rt_abi::arch::__twz_rt_upcall_entry),
267            0,
268            0,
269            0,
270            0.into(),
271            0.into(),
272            [UpcallOptions {
273                flags: UpcallFlags::empty(),
274                mode: UpcallMode::CallSelf,
275            }; UpcallInfo::NR_UPCALLS],
276        );
277        twizzler_abi::syscall::sys_thread_set_upcall(upcall_target).unwrap();
278        self.set_is_monitor();
279        self.init_allocator(init_info);
280        self.init_tls(init_info);
281        self.init_ctors(&init_info.ctors);
282    }
283
284    fn init_for_compartment(&self, init_info: &CompartmentInitInfo, entry_stack: *mut usize) {
285        let _start_1 = Instant::now();
286        unsafe {
287            preinit_unwrap(
288                monitor_api::set_comp_config(
289                    (init_info.comp_config_info as *const SharedCompConfig)
290                        .as_ref()
291                        .unwrap(),
292                )
293                .ok(),
294            );
295        }
296
297        let mut tg = TLS_GEN_MGR.lock();
298        let _start_2 = Instant::now();
299        let tls = tg.get_next_tls_info(None, || RuntimeThreadControl::new(0));
300        let _start_3 = Instant::now();
301        twizzler_abi::syscall::sys_thread_settls(preinit_unwrap(tls) as u64);
302        let _start_4 = Instant::now();
303        twizzler_abi::upcall::set_self_upcall_ptr(crate::arch::twz_rt_upcall_entry_c).unwrap();
304        let _start_5 = Instant::now();
305        libc_init_tcb(preinit_unwrap(tls));
306        self.init_core_thread(preinit_unwrap(tls));
307        if !unsafe { __mlibc_entry_from_rust.is_null() } {
308            let mlibc_entry_from_rust = unsafe {
309                std::mem::transmute::<_, extern "C" fn(*mut usize, *mut u8)>(
310                    __mlibc_entry_from_rust,
311                )
312            };
313            mlibc_entry_from_rust(entry_stack, core::ptr::null_mut());
314        }
315        let _start_6 = Instant::now();
316
317        if !init_info.ctor_set_array.is_null() && init_info.ctor_set_len != 0 {
318            let ctor_slice = unsafe {
319                core::slice::from_raw_parts(init_info.ctor_set_array, init_info.ctor_set_len)
320            };
321            self.init_ctors(ctor_slice);
322        }
323
324        /*
325        preinit_println!(
326            "init: {}us, get_tls: {}ms, set_tls: {}ms, set_upcall: {}ms, mlibc: {}ms, ctors: {}ms",
327            (_start_2 - _start_1).as_micros(),
328            (_start_3 - _start_2).as_millis(),
329            (_start_4 - _start_3).as_millis(),
330            (_start_5 - _start_4).as_millis(),
331            (_start_6 - _start_5).as_millis(),
332            _start_6.elapsed().as_millis()
333        )
334        */
335    }
336
337    fn init_ctors(&self, ctor_array: &[CtorSet]) {
338        for ctor in ctor_array {
339            unsafe {
340                if let Some(legacy_init) = ctor.legacy_init {
341                    (core::mem::transmute::<_, extern "C" fn()>(legacy_init))();
342                }
343                if !ctor.init_array.is_null() && ctor.init_array_len > 0 {
344                    let init_slice: &[usize] = core::slice::from_raw_parts(
345                        ctor.init_array as *const usize,
346                        ctor.init_array_len,
347                    );
348                    for call in init_slice.iter().cloned() {
349                        (core::mem::transmute::<_, extern "C" fn()>(call))();
350                    }
351                }
352            }
353        }
354    }
355
356    fn init_allocator(&self, info: &RuntimeInitInfo) {
357        for slot in &info.used_slots {
358            mark_slot_reserved(*slot);
359        }
360        self.register_bootstrap_alloc(info.bootstrap_alloc_slot);
361    }
362
363    fn init_tls(&self, info: &RuntimeInitInfo) {
364        let tls = info.tls_region.get_thread_pointer_value();
365        twizzler_abi::syscall::sys_thread_settls(tls as u64);
366    }

376use twizzler_rt_abi::core::rt0::__rust_entry_from_c;
377#[cfg(target_arch = "aarch64")]
378#[unsafe(no_mangle)]
379#[unsafe(naked)]
380pub unsafe extern "C" fn _start() {
381    core::arch::naked_asm!(
382        "b {entry}",
383        entry = sym __rust_entry_from_c,
384    );
385}
386
387#[cfg(target_arch = "x86_64")]
388#[unsafe(no_mangle)]
389#[unsafe(naked)]
390pub unsafe extern "C" fn _start() {
391    // Align the stack and jump to rust code. If we come back, trigger an exception.
392    core::arch::naked_asm!(
393        "and rsp, 0xfffffffffffffff0",
394        "call {entry}",
395        "ud2",
396        entry = sym __rust_entry_from_c,
397    );
398}
