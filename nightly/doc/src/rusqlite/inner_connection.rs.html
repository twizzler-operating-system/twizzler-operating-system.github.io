17pub struct InnerConnection {
18    pub db: *mut ffi::sqlite3,
19    // It's unsafe to call `sqlite3_close` while another thread is performing
20    // a `sqlite3_interrupt`, and vice versa, so we take this mutex during
21    // those functions. This protects a copy of the `db` pointer (which is
22    // cleared on closing), however the main copy, `db`, is unprotected.
23    // Otherwise, a long-running query would prevent calling interrupt, as
24    // interrupt would only acquire the lock after the query's completion.
25    interrupt_lock: Arc<Mutex<*mut ffi::sqlite3>>,
26    #[cfg(feature = "hooks")]
27    pub free_commit_hook: Option<unsafe fn(*mut std::os::raw::c_void)>,

29    pub free_rollback_hook: Option<unsafe fn(*mut std::os::raw::c_void)>,

66    pub fn open_with_flags(
67        c_path: &CStr,
68        mut flags: OpenFlags,
69        vfs: Option<&CStr>,
70    ) -> Result<Self> {
71        ensure_safe_sqlite_threading_mode()?;
72
73        let z_vfs = match vfs {
74            Some(c_vfs) => c_vfs.as_ptr(),
75            None => ptr::null(),
76        };
77
78        // turn on extended results code before opening database to have a better diagnostic if a failure happens
79        let exrescode = if version_number() >= 3_037_000 {
80            flags |= OpenFlags::SQLITE_OPEN_EXRESCODE;
81            true
82        } else {
83            false // flag SQLITE_OPEN_EXRESCODE is ignored by SQLite version < 3.37.0
84        };
85
86        unsafe {
87            let mut db: *mut ffi::sqlite3 = ptr::null_mut();
88            let r = ffi::sqlite3_open_v2(c_path.as_ptr(), &mut db, flags.bits(), z_vfs);
89            if r != ffi::SQLITE_OK {
90                let e = if db.is_null() {
91                    err!(r, "{}", c_path.to_string_lossy())
92                } else {
93                    let mut e = error_from_handle(db, r);
94                    if let Error::SqliteFailure(
95                        ffi::Error {
96                            code: ffi::ErrorCode::CannotOpen,
97                            ..
98                        },
99                        Some(msg),
100                    ) = e
101                    {
102                        e = err!(r, "{msg}: {}", c_path.to_string_lossy());
103                    }
104                    ffi::sqlite3_close(db);
105                    e
106                };
107
108                return Err(e);
109            }
110
111            // attempt to turn on extended results code; don't fail if we can't.
112            if !exrescode {
113                ffi::sqlite3_extended_result_codes(db, 1);
114            }
115
116            let r = ffi::sqlite3_busy_timeout(db, 5000);
117            if r != ffi::SQLITE_OK {
118                let e = error_from_handle(db, r);
119                ffi::sqlite3_close(db);
120                return Err(e);
121            }
122
123            Ok(Self::new(db, true))
124        }
125    }

137    pub fn close(&mut self) -> Result<()> {
138        if self.db.is_null() {
139            return Ok(());
140        }
141        self.remove_hooks();
142        self.remove_preupdate_hook();
143        let mut shared_handle = self.interrupt_lock.lock().unwrap();
144        assert!(
145            !self.owned || !shared_handle.is_null(),
146            "Bug: Somehow interrupt_lock was cleared before the DB was closed"
147        );
148        if !self.owned {
149            self.db = ptr::null_mut();
150            return Ok(());
151        }
152        unsafe {
153            let r = ffi::sqlite3_close(self.db);
154            // Need to use _raw because _guard has a reference out, and
155            // decode_result takes &mut self.
156            let r = decode_result_raw(self.db, r);
157            if r.is_ok() {
158                *shared_handle = ptr::null_mut();
159                self.db = ptr::null_mut();
160            }
161            r
162        }
163    }

180    pub unsafe fn load_extension(
181        &self,
182        dylib_path: &Path,
183        entry_point: Option<&str>,
184    ) -> Result<()> {
185        let dylib_str = super::path_to_cstring(dylib_path)?;
186        let mut errmsg: *mut c_char = ptr::null_mut();
187        let r = if let Some(entry_point) = entry_point {
188            let c_entry = crate::str_to_cstring(entry_point)?;
189            ffi::sqlite3_load_extension(self.db, dylib_str.as_ptr(), c_entry.as_ptr(), &mut errmsg)
190        } else {
191            ffi::sqlite3_load_extension(self.db, dylib_str.as_ptr(), ptr::null(), &mut errmsg)
192        };
193        if r == ffi::SQLITE_OK {
194            Ok(())
195        } else {
196            let message = super::errmsg_to_string(errmsg);
197            ffi::sqlite3_free(errmsg.cast::<std::os::raw::c_void>());
198            Err(crate::error::error_from_sqlite_code(r, Some(message)))
199        }
200    }

207    pub fn prepare<'a>(
208        &mut self,
209        conn: &'a Connection,
210        sql: &str,
211        flags: PrepFlags,
212    ) -> Result<Statement<'a>> {
213        let mut c_stmt: *mut ffi::sqlite3_stmt = ptr::null_mut();
214        let (c_sql, len, _) = str_for_sqlite(sql.as_bytes())?;
215        let mut c_tail: *const c_char = ptr::null();
216        #[cfg(not(feature = "unlock_notify"))]
217        let r = unsafe { self.prepare_(c_sql, len, flags, &mut c_stmt, &mut c_tail) };
218        #[cfg(feature = "unlock_notify")]
219        let r = unsafe {
220            use crate::unlock_notify;
221            let mut rc;
222            loop {
223                rc = self.prepare_(c_sql, len, flags, &mut c_stmt, &mut c_tail);
224                if !unlock_notify::is_locked(self.db, rc) {
225                    break;
226                }
227                rc = unlock_notify::wait_for_unlock_notify(self.db);
228                if rc != ffi::SQLITE_OK {
229                    break;
230                }
231            }
232            rc
233        };
234        // If there is an error, *ppStmt is set to NULL.
235        if r != ffi::SQLITE_OK {
236            return Err(unsafe { error_with_offset(self.db, r, sql) });
237        }
238        // If the input text contains no SQL (if the input is an empty string or a
239        // comment) then *ppStmt is set to NULL.
240        let tail = if c_tail.is_null() {
241            0
242        } else {
243            let n = (c_tail as isize) - (c_sql as isize);
244            if n <= 0 || n >= len as isize {
245                0
246            } else {
247                n as usize
248            }
249        };
250        Ok(Statement::new(conn, unsafe {
251            RawStatement::new(c_stmt, tail)
252        }))
253    }

310    pub fn is_busy(&self) -> bool {
311        let db = self.db();
312        unsafe {
313            let mut stmt = ffi::sqlite3_next_stmt(db, ptr::null_mut());
314            while !stmt.is_null() {
315                if ffi::sqlite3_stmt_busy(stmt) != 0 {
316                    return true;
317                }
318                stmt = ffi::sqlite3_next_stmt(db, stmt);
319            }
320        }
321        false
322    }

342            -1 => Err(err!(
343                ffi::SQLITE_MISUSE,
344                "{db_name:?} is not the name of a database"

346            _ => Err(err!(r, "Unexpected result")),

365            -1 => Err(err!(
366                ffi::SQLITE_MISUSE,
367                "{db_name:?} is not the name of a valid schema"

369            _ => Err(err!(r, "Unexpected result")),

411// threading mode checks are not necessary (and do not work) on target
412// platforms that do not have threading (such as webassembly)
413#[cfg(target_arch = "wasm32")]
414fn ensure_safe_sqlite_threading_mode() -> Result<()> {
415    Ok(())
416}
417
418#[cfg(not(any(target_arch = "wasm32")))]
419fn ensure_safe_sqlite_threading_mode() -> Result<()> {
420    // Ensure SQLite was compiled in threadsafe mode.
421    if unsafe { ffi::sqlite3_threadsafe() == 0 } {
422        return Err(Error::SqliteSingleThreadedMode);
423    }
424
425    // Now we know SQLite is _capable_ of being in Multi-thread of Serialized mode,
426    // but it's possible someone configured it to be in Single-thread mode
427    // before calling into us. That would mean we're exposing an unsafe API via
428    // a safe one (in Rust terminology).
429    //
430    // We can ask SQLite for a mutex and check for
431    // the magic value 8. This isn't documented, but it's what SQLite
432    // returns for its mutex allocation function in Single-thread mode.
433    const SQLITE_SINGLETHREADED_MUTEX_MAGIC: usize = 8;
434    let is_singlethreaded = unsafe {
435        let mutex_ptr = ffi::sqlite3_mutex_alloc(0);
436        let is_singlethreaded = mutex_ptr as usize == SQLITE_SINGLETHREADED_MUTEX_MAGIC;
437        ffi::sqlite3_mutex_free(mutex_ptr);
438        is_singlethreaded
439    };
440    if is_singlethreaded {
441        Err(Error::SqliteSingleThreadedMode)
442    } else {
443        Ok(())
444    }
