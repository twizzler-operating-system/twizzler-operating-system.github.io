9/// Specifications for an object-copy from a source object. The specified ranges are
10/// source:[src_start, src_start + len) copied to <some unspecified destination object>:[dest_start,
11/// dest_start + len). Each range must start within an object, and end within the object.
12pub struct ObjectSource {
13    /// The ID of the source object, or zero for filling destination with zero.
14    pub id: ObjID,
15    /// The offset into the source object to start the copy. If id is zero, this field is reserved
16    /// for future use.
17    pub src_start: u64,
18    /// The offset into the dest object to start the copy or zero.
19    pub dest_start: u64,
20    /// The length of the copy or zero.
21    pub len: usize,
22}
23
24impl ObjectSource {
25    /// Construct a new ObjectSource.
26    pub fn new_copy(id: ObjID, src_start: u64, dest_start: u64, len: usize) -> Self {
27        Self {
28            id,
29            src_start,
30            dest_start,
31            len,
32        }
33    }
34
35    /// Construct a new ObjectSource.
36    pub fn new_zero(dest_start: u64, len: usize) -> Self {
37        Self {
38            id: ObjID::new(0),
39            src_start: 0,
40            dest_start,
41            len,
42        }
43    }
44}

48/// The backing memory type for this object. Currently doesn't do anything.
49pub enum BackingType {
50    /// The default, let the kernel decide based on the [LifetimeType] of the object.
51    #[default]

53}

57/// The base lifetime type of the object. Note that this does not ensure that the object is stored
58/// in a specific type of memory, the kernel is allowed to migrate objects with the Normal
59/// [BackingType] as it sees fit. For more information on object lifetime, see [the book](https://twizzler-operating-system.github.io/nightly/book/object_lifetime.html).
60pub enum LifetimeType {
61    /// This object is volatile, and is expected to be deleted after a power cycle.
62    #[default]
63    Volatile = 0,
64    /// This object is persistent, and should be deleted only after an explicit delete call.
65    Persistent = 1,
66}

125/// Create an object, returning either its ID or an error.
126pub fn sys_object_create(
127    create: ObjectCreate,
128    sources: &[ObjectSource],
129    ties: &[CreateTieSpec],
130) -> Result<ObjID> {
131    let args = [
132        &create as *const ObjectCreate as u64,
133        sources.as_ptr() as u64,
134        sources.len() as u64,
135        ties.as_ptr() as u64,
136        ties.len() as u64,
137    ];
138    let (code, val) = unsafe { raw_syscall(Syscall::ObjectCreate, &args) };
139    convert_codes_to_result(
140        code,
141        val,
142        |c, _| c == 0,
143        |x, y| crate::object::ObjID::from_parts([x, y]),
144        twzerr,
145    )
146}
