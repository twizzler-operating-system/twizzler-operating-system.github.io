12/// A capability that represents authorization for a [Security Context](`crate::sec_ctx::SecCtx`) to
13/// access an object.
14///
15/// Capabilities are stored inside [`crate::sec_ctx::SecCtx`], and are authenticated
16/// using cryptographic signatures. When accessing an object for the first time,
17/// the kernel searches through the attached [Security Context](`crate::sec_ctx::SecCtx`) for
18/// a usable capability. If none found it will look through inactive contexts for a valid
19/// capability and then procedes to verify its signature in order to grant access rights.
20///
21///
22/// # Fields
23///
24/// * `target` - The object ID this capability grants access to
25/// * `accessor` - The security context ID in which this capability resides
26/// * `protections` - The specific access rights this capability grants
27/// * `flags` - Specifies the cryptographic primitives used to form the signature
28/// * `gates` - Allows access into an object in a specified range
29/// * `revocation` - Specifies when the capability is invalid
30/// * `signature` - the signature of the capability
31///
32/// # Examples
33///
34/// ```
35/// // Example of creating and using a capability
36/// todo
37/// ```
38#[derive(Clone, Copy, PartialEq, Eq, Debug)]
39pub struct Cap {
40    /// Object ID this capability grants access to
41    pub target: ObjID,
42
43    /// Security context ID in which this capability resides
44    pub accessor: ObjID,
45
46    /// Specific access rights this capability grants
47    pub protections: Protections,
48
49    /// Cryptographic configuration for capability validation
50    flags: CapFlags,
51
52    /// Additional constraints on when this capability can be used
53    gates: Gates,
54
55    /// Specifies when this capability is invalid, i.e. expiration.
56    pub revocation: Revoc,
57
58    /// The signature inside the capability
59    sig: Signature,
60}

62const CAP_SERIALIZED_LEN: usize = 78;

64impl Cap {
65    /// creating a new capability, revoc specified in expiration data in ns from unix epoch
66    pub fn new(
67        target: ObjID,
68        accessor: ObjID,
69        prots: Protections,
70        target_priv_key: SigningKey,
71        revocation: Revoc,
72        gates: Gates,
73        hashing_algo: HashingAlgo,
74        signing_scheme: SigningScheme,
75    ) -> Result<Self, SecurityError> {
76        let cf_hashing_algo: CapFlags = hashing_algo.into();
77        let cf_signing_scheme: CapFlags = signing_scheme.into();
78
79        let flags = cf_hashing_algo | cf_signing_scheme; // set flags
80
81        #[cfg(feature = "log")]
82        debug!(
83            "Using flags: {} to create capability for target: {:?}",
84            flags, target
85        );
86
87        let hash_arr = Cap::serialize(accessor, target, prots, flags, revocation, gates);
88
89        let sig = match hashing_algo {
90            HashingAlgo::Blake3 => {
91                // unimplemented!("running into problems with blake3 compilation on aarch64");
92                let hash = blake3::hash(&hash_arr);
93                target_priv_key.sign(hash.as_bytes())?
94            }
95            HashingAlgo::Sha256 => {
96                let mut hasher = Sha256::new();
97                hasher.update(hash_arr);
98                let hash = hasher.finalize();
99                target_priv_key.sign(hash.as_slice())?
100            }
101        };
102
103        Ok(Cap {
104            accessor,
105            target,
106            protections: prots,
107            flags,
108            revocation,
109            gates,
110            sig,
111        })
112    }

115
116    pub fn verify_sig(&self, verifying_key: VerifyingKey) -> Result<(), SecurityError> {
117        let hash_arr = Self::serialize(
118            self.accessor,
119            self.target,
120            self.protections,
121            self.flags,
122            self.revocation,
123            self.gates,
124        );
125
126        let hash_algo: HashingAlgo = self.flags.try_into()?;
127
128        match hash_algo {
129            HashingAlgo::Blake3 => {
130                // #[cfg(feature = "log")]
131                // error!("running into problems with blake3 compilation on aarch64");
132                // unimplemented!("running into problems with blake3 compilation on aarch64");
133                let hash = blake3::hash(&hash_arr);
134                let bind = hash.as_bytes();
135                verifying_key.verify(bind.as_slice(), &self.sig)
136            }
137            HashingAlgo::Sha256 => {
138                #[cfg(feature = "log")]
139                debug!("Hashing via Sha256");
140                let mut hasher = sha2::Sha256::new();
141                hasher.update(&hash_arr);
142                let result = hasher.finalize();
143                verifying_key.verify(result.as_slice(), &self.sig)
144            }
145        }
146    }

150    pub fn check_gate(&self, offset: u64, length: u64, align: u64) -> Result<(), SecurityError> {
151        // the offset and length fields specify a region within the object. when the kernel switches

153        // it checks to see if the instruction pointer is in a valid gate for the object it points
154        // to. The instruction pointer must reside within the region specified by offset and
155        // length and must be aligned on a value specified by align.
156
157        //  assuming the layout is something like
158        // ||||||||||||||||||||||||||||||||||||||||||||||||||||
159        // offset |                                       | length
160        //        {                                       }
161        // the proposed offset must lay in this region
162
163        //TODO: this needs to be fixed so that any 'chunk' inside of the reigion is valid too
164        // if self.gates.offset < offset || offset > self.gates.offset + length {
165        //     return Err(SecurityError::OutsideBounds);
166        // }
167
168        //TODO: make sure this is correct
169        if !(offset + length < self.gates.length && offset > self.gates.offset) {
170            return Err(SecurityError::GateDenied);
171        }
172
173        //NOTE: not completely sure this is how you check alignment.
174        if self.gates.align != align {
175            return Err(SecurityError::InvalidGate);
176        }

181    /// returns all contents other than sig as a buffer ready to hash
182    fn serialize(
183        accessor: ObjID,
184        target: ObjID,
185        prots: Protections,
186        flags: CapFlags,
187        revocation: Revoc,
188        gates: Gates,
189    ) -> [u8; CAP_SERIALIZED_LEN] {
190        let mut hash_arr: [u8; CAP_SERIALIZED_LEN] = [0; CAP_SERIALIZED_LEN];
191        hash_arr[0..16].copy_from_slice(&accessor.raw().to_le_bytes());
192        hash_arr[16..32].copy_from_slice(&target.raw().to_le_bytes());
193        hash_arr[32..36].copy_from_slice(&prots.bits().to_le_bytes());
194        hash_arr[36..38].copy_from_slice(&flags.bits().to_le_bytes());
195        hash_arr[38..54].copy_from_slice(&revocation.to_bytes());
196        hash_arr[54..62].copy_from_slice(&gates.offset.to_le_bytes());
197        hash_arr[62..70].copy_from_slice(&gates.length.to_le_bytes());
198        hash_arr[70..78].copy_from_slice(&gates.align.to_le_bytes());
199        hash_arr
200    }
