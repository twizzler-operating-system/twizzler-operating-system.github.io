10impl RawTwzError {
11    pub fn category(&self) -> ErrorCategory {
12        let cat = (self.0 & bindings::ERROR_CATEGORY_MASK) >> bindings::ERROR_CATEGORY_SHIFT;
13        match cat as u16 {
14            bindings::GENERIC_ERROR => ErrorCategory::Generic,
15            bindings::ARGUMENT_ERROR => ErrorCategory::Argument,
16            bindings::RESOURCE_ERROR => ErrorCategory::Resource,
17            bindings::NAMING_ERROR => ErrorCategory::Naming,
18            bindings::OBJECT_ERROR => ErrorCategory::Object,
19            bindings::IO_ERROR => ErrorCategory::Io,
20            _ => ErrorCategory::Uncategorized,
21        }
22    }
23
24    pub fn code(&self) -> u16 {
25        ((self.0 & bindings::ERROR_CODE_MASK) >> bindings::ERROR_CODE_SHIFT) as u16
26    }
27
28    pub fn from_parts(cat: u16, code: u16) -> Self {
29        let cat = ((cat as u64) << bindings::ERROR_CATEGORY_SHIFT) & bindings::ERROR_CATEGORY_MASK;
30        let code = ((code as u64) << bindings::ERROR_CODE_SHIFT) & bindings::ERROR_CODE_MASK;
31        Self(cat | code)
32    }
33
34    pub fn error(&self) -> TwzError {
35        match self.category() {
36            ErrorCategory::Uncategorized => TwzError::Uncategorized(self.code()),
37            ErrorCategory::Generic => GenericError::twz_error_from_code(self.code()),
38            ErrorCategory::Argument => ArgumentError::twz_error_from_code(self.code()),
39            ErrorCategory::Resource => ResourceError::twz_error_from_code(self.code()),
40            ErrorCategory::Naming => NamingError::twz_error_from_code(self.code()),
41            ErrorCategory::Object => ObjectError::twz_error_from_code(self.code()),
42            ErrorCategory::Io => IoError::twz_error_from_code(self.code()),
43        }
44    }
45
46    pub fn new(raw: bindings::twz_error) -> Self {
47        Self(raw)
48    }
49
50    pub fn is_success(&self) -> bool {
51        self.code() == bindings::SUCCESS
52    }
53
54    pub fn raw(&self) -> bindings::twz_error {
55        self.0
56    }
57
58    pub fn success() -> Self {
59        Self(bindings::SUCCESS as u64)
60    }
61
62    pub fn result(&self) -> Result<(), TwzError> {
63        if self.is_success() {
64            Ok(())
65        } else {
66            Err(self.error())
67        }
68    }
69}

72pub enum TwzError {
73    Uncategorized(u16),
74    Generic(GenericError),
75    Argument(ArgumentError),
76    Resource(ResourceError),
77    Object(ObjectError),
78    Naming(NamingError),
79    Io(IoError),
80}

82impl TwzError {
83    pub const NOT_SUPPORTED: Self = Self::Generic(GenericError::NotSupported);
84    pub const TIMED_OUT: Self = Self::Generic(GenericError::TimedOut);
85    pub const WOULD_BLOCK: Self = Self::Generic(GenericError::WouldBlock);
86    pub const INVALID_ARGUMENT: Self = Self::Argument(ArgumentError::InvalidArgument);
87
88    pub fn category(&self) -> ErrorCategory {
89        match self {
90            TwzError::Uncategorized(_) => ErrorCategory::Uncategorized,
91            TwzError::Generic(_) => ErrorCategory::Generic,
92            TwzError::Argument(_) => ErrorCategory::Argument,
93            TwzError::Resource(_) => ErrorCategory::Resource,
94            TwzError::Object(_) => ErrorCategory::Object,
95            TwzError::Io(_) => ErrorCategory::Io,
96            TwzError::Naming(_) => ErrorCategory::Naming,
97        }
98    }
99
100    pub fn raw(&self) -> bindings::twz_error {
101        let cat = self.category().raw();
102        let code = self.code();
103        RawTwzError::from_parts(cat, code).raw()
104    }
105
106    pub fn code(&self) -> u16 {
107        match self {
108            TwzError::Uncategorized(code) => *code,
109            TwzError::Generic(generic_error) => generic_error.code(),
110            TwzError::Argument(argument_error) => argument_error.code(),
111            TwzError::Resource(resource_error) => resource_error.code(),
112            TwzError::Object(object_error) => object_error.code(),
113            TwzError::Io(io_error) => io_error.code(),
114            TwzError::Naming(naming_error) => naming_error.code(),
115        }
116    }
117}
