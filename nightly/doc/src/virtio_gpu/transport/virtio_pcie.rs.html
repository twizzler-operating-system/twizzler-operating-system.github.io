1// Virtio vendor specific PCI Capability
2#[repr(C)]
3pub struct VirtioPciCap {
4    pub cap_vndr: u8,            /* Generic PCI field: PCI_CAP_ID_VNDR */
5    pub cap_next: u8,            /* Generic PCI field: next ptr. */
6    pub cap_len: u8,             /* Generic PCI field: capability length */
7    pub cfg_type: VirtioCfgType, /* Identifies the structure. */
8    pub bar: u8,                 /* Where to find it. */
9    pub id: u8,                  /* Multiple capabilities of the same type */
10    pub padding: [u8; 2],        /* Pad to full dword. */
11    pub offset: u32,             /* Offset within bar. */
12    pub length: u32,             /* Length of the structure, in bytes. */
13}
