21async fn handle_page_data_request_task(
22    ctx: &'static PagerContext,
23    qid: u32,
24    id: ObjID,
25    mut req_range: ObjectRange,
26    flags: PagerFlags,
27) {
28    static COUNT: AtomicU64 = AtomicU64::new(0);
29    static PCOUNT: AtomicU64 = AtomicU64::new(0);
30    let prefetch = flags.contains(PagerFlags::PREFETCH);
31
32    if req_range.start == 0 {
33        req_range.start = PAGE;
34    }
35    let start_time = Instant::now();
36    if prefetch {
37        tracing::info!("STARTING {}: {:?} {:?}", id, req_range, flags);
38        if let Ok(len) = blocking::unblock(move || ctx.paged_ostore(None)?.len(id.raw())).await {
39            tracing::debug!(
40                "==> prefetch request reduce len: {} -> {}",
41                req_range.end,
42                len
43            );
44            req_range.end = len.next_multiple_of(PAGE) + PAGE;
45        }
46        PCOUNT.fetch_add(1, Ordering::SeqCst);
47    } else {
48        COUNT.fetch_add(1, Ordering::SeqCst);
49    }
50
51    let total = req_range.pages().count() as u64;
52    let mut count = 0;
53    while count < total {
54        tracing::trace!(
55            "reading {} page {} of {} (pre = {})",
56            id,
57            count,
58            total,
59            prefetch
60        );
61        let range = ObjectRange::new(req_range.start + count * PAGE, req_range.end);
62        let pages = match ctx
63            .data
64            .fill_mem_pages_partial(ctx, id, range)
65            .await
66            .inspect_err(|e| tracing::warn!("page data request failed: {}", e))
67        {
68            Ok(pages) => pages,
69            Err(e) => {
70                let comp = CompletionToKernel::new(
71                    KernelCompletionData::Error(RawTwzError::new(e.raw())),
72                    KernelCompletionFlags::DONE,
73                );
74                ctx.notify_kernel(qid, comp).await;
75                return;
76            }
77        };
78
79        let thiscount = pages
80            .iter()
81            .fold(0u64, |acc, x| acc + (x.range.end - x.range.start) / PAGE);
82
83        // try to compress page ranges
84        let runs = crate::helpers::consecutive_slices(pages.as_slice());
85        let mut acc = 0;
86        let comps = runs
87            .map(|run| {
88                let start = run[0];
89                let last = run.last().unwrap();
90                let flags = if start.is_wired() {
91                    PageFlags::WIRED
92                } else {
93                    PageFlags::empty()
94                };
95                let phys_range = PhysRange {
96                    start: start.range.start,
97                    end: last.range.end,
98                };
99
100                let start = req_range.start + (count + acc as u64) * PAGE;
101                let range = ObjectRange::new(start, start + phys_range.len() as u64);
102
103                acc += phys_range.pages().count();
104                CompletionToKernel::new(
105                    KernelCompletionData::PageDataCompletion(id, range, phys_range, flags),
106                    KernelCompletionFlags::empty(),
107                )
108            })
109            .collect::<Vec<_>>();
110
111        tracing::trace!(
112            "sending {} kernel notifs for {} ({} pages)",
113            comps.len(),
114            id,
115            thiscount
116        );
117        for comp in comps.iter() {
118            ctx.notify_kernel(qid, *comp).await;
119        }
120        count += thiscount;
121    }
122    if prefetch {
123        PCOUNT.fetch_sub(1, Ordering::SeqCst);
124    } else {
125        COUNT.fetch_sub(1, Ordering::SeqCst);
126    }
127    if prefetch {
128        tracing::info!(
129            "COMPLETED: {} {:?} in {} ms, {}:{} remaining",
130            id,
131            req_range,
132            start_time.elapsed().as_millis(),
133            COUNT.load(Ordering::SeqCst),
134            PCOUNT.load(Ordering::SeqCst),
135        );
136    }
137
138    let done = CompletionToKernel::new(KernelCompletionData::Okay, KernelCompletionFlags::DONE);
139    ctx.notify_kernel(qid, done).await;
140}

142async fn handle_page_data_request(
143    ctx: &'static PagerContext,
144    qid: u32,
145    id: ObjID,
146    req_range: ObjectRange,
147    flags: PagerFlags,
148) -> Vec<CompletionToKernel> {
149    tracing::debug!(
150        "{}: {:?} {} pages",
151        id,
152        req_range,
153        req_range.pages().count()
154    );
155    let _task = EXECUTOR
156        .get()
157        .unwrap()
158        .spawn(async move {
159            handle_page_data_request_task(ctx, qid, id, req_range, flags).await;
160        })
161        .detach();
162    vec![]
163}

165async fn object_info_req(ctx: &'static PagerContext, id: ObjID) -> Result<ObjectInfo> {
166    ctx.data.lookup_object(ctx, id).await
167}

169async fn handle_sync_region(
170    ctx: &'static PagerContext,
171    info: ObjectEvictInfo,
172) -> CompletionToKernel {
173    if !info.flags.contains(ObjectEvictFlags::SYNC) {
174        return CompletionToKernel::new(
175            KernelCompletionData::Error(TwzError::NOT_SUPPORTED.into()),
176            KernelCompletionFlags::DONE,
177        );
178    }
179    ctx.data.sync_region(ctx, &info).await
180}

182pub async fn handle_kernel_request(
183    ctx: &'static PagerContext,
184    qid: u32,
185    request: RequestFromKernel,
186) -> Vec<CompletionToKernel> {
187    tracing::trace!("handling kernel request {:?}", request);
188
189    let data = match request.cmd() {
190        KernelCommand::PageDataReq(obj_id, range, flags) => {
191            return handle_page_data_request(ctx, qid, obj_id, range, flags).await;
192        }
193        KernelCommand::ObjectInfoReq(obj_id) => match object_info_req(ctx, obj_id).await {
194            Ok(info) => KernelCompletionData::ObjectInfoCompletion(obj_id, info),
195            Err(e) => KernelCompletionData::Error(e.into()),
196        },
197
198        KernelCommand::ObjectDel(obj_id) => {
199            unblock(move || {
200                let res = ctx
201                    .paged_ostore(None)
202                    .map(|po| po.delete_object(obj_id.raw()));
203                match res {
204                    Ok(_) => {
205                        let _ = ctx.paged_ostore(None).map(|po| {
206                            po.flush()
207                                .inspect_err(|e| tracing::warn!("failed to advance epoch: {}", e))
208                        });
209                        KernelCompletionData::Okay
210                    }
211                    Err(e) => KernelCompletionData::Error(TwzError::from(e).into()),
212                }
213            })
214            .await
215        }
216        KernelCommand::ObjectCreate(id, object_info) => {
217            blocking::unblock(move || {
218                let _ = ctx.paged_ostore(None).map(|po| po.delete_object(id.raw()));
219                match ctx.paged_ostore(None).map(|po| po.create_object(id.raw())) {
220                    Ok(_) => {
221                        let mut buffer = [0; 0x1000];
222                        let meta = MetaInfo {
223                            nonce: Nonce(object_info.nonce),
224                            kuid: object_info.kuid,
225                            default_prot: object_info.def_prot,
226                            flags: MetaFlags::empty(),
227                            fotcount: 0,
228                            extcount: 0,
229                        };
230                        unsafe fn any_as_u8_slice<T: Sized>(p: &T) -> &[u8] {
231                            ::core::slice::from_raw_parts(
232                                (p as *const T) as *const u8,
233                                ::core::mem::size_of::<T>(),
234                            )
235                        }
236                        unsafe {
237                            buffer[0..size_of::<MetaInfo>()]
238                                .copy_from_slice(any_as_u8_slice(&meta));
239                        }
240                        ctx.paged_ostore(None)
241                            .unwrap()
242                            .write_object(id.raw(), 0, &buffer)
243                            .unwrap();
244
245                        KernelCompletionData::ObjectInfoCompletion(id, object_info)
246                    }
247                    Err(e) => {
248                        tracing::warn!("failed to create object {}: {}", id, e);
249                        KernelCompletionData::Error(TwzError::from(e).into())
250                    }
251                }
252            })
253            .await
254        }
255        KernelCommand::DramPages(phys_range) => {
256            tracing::debug!("tracking {} KB memory", phys_range.len() / 1024);
257            ctx.data.add_memory_range(phys_range);
258            KernelCompletionData::Okay
259        }
260        KernelCommand::ObjectEvict(info) => {
261            return vec![handle_sync_region(ctx, info).await];
262        }
263    };
264
265    tracing::debug!("done; sending response: {:?}", data);
266    vec![CompletionToKernel::new(data, KernelCompletionFlags::DONE)]
267}
