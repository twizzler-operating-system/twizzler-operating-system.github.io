57const READY: u64 = 1;
58const DROPPED: u64 = 2;
59
60pub struct NvmeRequest {
61    cmd: CommonCommand,
62    ready: UnsafeCell<MaybeUninit<CommonCompletion>>,
63    flags: AtomicU64,
64}

66impl<'a> Drop for InflightRequest<'a> {
67    fn drop(&mut self) {
68        let requests = &mut self.req.inner.lock().unwrap().requests;
69        let entry = requests.get(self.id as usize).unwrap();
70        if entry.flags.fetch_or(DROPPED, Ordering::SeqCst) & READY != 0 {
71            requests.remove(self.id as usize);
72        }
73    }
74}

76impl<'a> TwizzlerWaitable for InflightRequest<'a> {
77    fn wait_item_read(&self) -> twizzler_abi::syscall::ThreadSyncSleep {
78        let requests = &self.req.inner.lock().unwrap().requests;
79        let req = requests.get(self.id as usize).unwrap();
80        ThreadSyncSleep::new(
81            ThreadSyncReference::Virtual(&req.flags),
82            0,
83            twizzler_abi::syscall::ThreadSyncOp::Equal,
84            ThreadSyncFlags::empty(),
85        )
86    }
87
88    fn wait_item_write(&self) -> twizzler_abi::syscall::ThreadSyncSleep {
89        self.wait_item_read()
90    }
91}

103impl NvmeRequesterInner {
104    pub fn new(
105        subq: SubmissionQueue,
106        comq: CompletionQueue,
107        sub_bell: *mut u32,
108        com_bell: *mut u32,
109        bar_obj: MmioObject,
110        sub_dma: NvmeDmaSliceRegion<CommonCommand>,
111        com_dma: NvmeDmaSliceRegion<CommonCompletion>,
112    ) -> Self {
113        Self {
114            subq,
115            comq,
116            sub_bell,
117            com_bell,
118            requests: Slab::new(),
119            _sub_dma: sub_dma,
120            _com_dma: com_dma,
121            _bar_obj: bar_obj,
122        }
123    }
124
125    #[inline]
126    fn sub_bell(&self) -> VolatilePtr<'_, u32> {
127        unsafe { VolatilePtr::new(NonNull::new(self.sub_bell).unwrap()) }
128    }
129
130    #[inline]
131    fn com_bell(&self) -> VolatilePtr<'_, u32> {
132        unsafe { VolatilePtr::new(NonNull::new(self.com_bell).unwrap()) }
133    }
134
135    #[inline]
136    pub fn get_completion(&mut self) -> Option<CommonCompletion> {
137        let Some((bell, resp)) = self.comq.get_completion::<CommonCompletion>() else {
138            return None;
139        };
140        self.subq.update_head(resp.new_sq_head());
141        self.com_bell().write(bell as u32);
142        let id: u16 = resp.command_id().into();
143        let entry = self.requests.get(id as usize).unwrap();
144        unsafe { entry.ready.get().as_mut().unwrap().write(resp) };
145        if entry.flags.fetch_or(READY, Ordering::SeqCst) & DROPPED != 0 {
146            self.requests.remove(id as usize);
147        } else {
148            let _ = twizzler_abi::syscall::sys_thread_sync(
149                &mut [ThreadSync::new_wake(ThreadSyncWake::new(
150                    ThreadSyncReference::Virtual(&entry.flags),
151                    usize::MAX,
152                ))],
153                None,
154            );
155        }
156
157        Some(resp)
158    }
159
160    #[inline]
161    pub fn submit(&mut self, mut cmd: CommonCommand) -> Option<u16> {
162        let entry = self.requests.vacant_entry();
163        let id = entry.key() as u16;
164        cmd.set_cid(id.into());
165        entry.insert(NvmeRequest::new(cmd));
166        let entry = self.requests.get(id as usize)?;
167        if let Some(tail) = self.subq.submit(&entry.cmd) {
168            self.sub_bell().write(tail as u32);
169            Some(id)
170        } else {
171            self.requests.remove(id as usize);
172            None
173        }
174    }
175
176    #[inline]
177    pub fn poll(&self, inflight: &InflightRequest) -> std::io::Result<CommonCompletion> {
178        let Some(entry) = self.requests.get(inflight.id as usize) else {
179            return Err(ErrorKind::Other.into());
180        };
181        if entry.flags.load(Ordering::SeqCst) & READY != 0 {
182            Ok(unsafe { entry.ready.get().as_ref().unwrap().assume_init_read() })
183        } else {
184            Err(ErrorKind::WouldBlock.into())
185        }
186    }
187}

189impl NvmeRequester {
190    pub fn new(
191        subq: SubmissionQueue,
192        comq: CompletionQueue,
193        sub_bell: *mut u32,
194        com_bell: *mut u32,
195        bar_obj: MmioObject,
196        sub_dma: NvmeDmaSliceRegion<CommonCommand>,
197        com_dma: NvmeDmaSliceRegion<CommonCompletion>,
198    ) -> Self {
199        Self {
200            inner: Mutex::new(NvmeRequesterInner::new(
201                subq, comq, sub_bell, com_bell, bar_obj, sub_dma, com_dma,
202            )),
203            cv: Condvar::new(),
204        }
205    }
206
207    #[inline]
208    pub fn submit(&self, cmd: CommonCommand) -> Option<InflightRequest<'_>> {
209        let id = self.inner.lock().unwrap().submit(cmd)?;
210        Some(InflightRequest { req: self, id })
211    }
212
213    #[inline]
214    pub fn submit_wait(
215        &self,
216        cmd: CommonCommand,
217        timeout: Option<Duration>,
218    ) -> Option<InflightRequest<'_>> {
219        let mut inner = self.inner.lock().unwrap();
220        loop {
221            if let Some(id) = inner.submit(cmd) {
222                return Some(InflightRequest { req: self, id });
223            }
224            if let Some(timeout) = timeout {
225                let (guard, to) = self.cv.wait_timeout(inner, timeout).unwrap();
226                if to.timed_out() {
227                    return None;
228                }
229                inner = guard;
230            } else {
231                inner = self.cv.wait(inner).unwrap();
232            }
233        }
234    }
235
236    pub fn poll(&self, inflight: &InflightRequest) -> std::io::Result<CommonCompletion> {
237        self.inner.lock().unwrap().poll(inflight)
238    }
239
240    pub fn check_completions(&self) -> bool {
241        let mut inner = self.inner.lock().unwrap();
242        let mut more = false;
243        while let Some(_) = inner.get_completion() {
244            more = true;
245        }
246        if more {
247            self.cv.notify_all();
248        }
249        more
250    }
251
252    pub fn get_completion(&self) -> Option<CommonCompletion> {
253        let cc = self.inner.lock().unwrap().get_completion();
254        if cc.is_some() {
255            self.cv.notify_one();
256        }
257        cc
258    }
