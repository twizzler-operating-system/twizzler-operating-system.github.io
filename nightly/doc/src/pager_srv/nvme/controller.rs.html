56fn init_controller(mut device: Device, mut dma_pool: DmaPool) -> std::io::Result<NvmeController> {
57    let bar = device.get_mmio(1).unwrap();
58    let mut reg = unsafe {
59        bar.get_mmio_offset_mut::<nvme::ds::controller::properties::ControllerProperties>(0)
60    };
61    let reg = reg.as_mut_ptr();
62
63    let _int = device
64        .allocate_interrupt(0)
65        .expect("failed to allocate interrupt");
66    let config = ControllerConfig::new();
67    map_field!(reg.configuration).write(config);
68
69    while map_field!(reg.status).read().ready() {
70        core::hint::spin_loop();
71    }
72
73    let aqa = nvme::ds::controller::properties::aqa::AdminQueueAttributes::new()
74        .with_completion_queue_size(ADMIN_QUEUE_LEN - 1)
75        .with_submission_queue_size(ADMIN_QUEUE_LEN - 1);
76    map_field!(reg.admin_queue_attr).write(aqa);
77
78    let saq = dma_pool
79        .allocate_array(
80            ADMIN_QUEUE_LEN as usize,
81            nvme::ds::queue::subentry::CommonCommand::default(),
82        )
83        .unwrap();
84    let caq = dma_pool
85        .allocate_array(
86            ADMIN_QUEUE_LEN as usize,
87            nvme::ds::queue::comentry::CommonCompletion::default(),
88        )
89        .unwrap();
90
91    let mut saq = NvmeDmaSliceRegion::new(saq);
92    let mut caq = NvmeDmaSliceRegion::new(caq);
93
94    let cpin = caq.dma_region_mut().pin().unwrap();
95    let spin = saq.dma_region_mut().pin().unwrap();
96
97    assert_eq!(cpin.len(), 1);
98    assert_eq!(spin.len(), 1);
99
100    let cpin_addr = cpin[0].addr();
101    let spin_addr = spin[0].addr();
102
103    map_field!(reg.admin_comqueue_base_addr).write(cpin_addr.into());
104    map_field!(reg.admin_subqueue_base_addr).write(spin_addr.into());
105
106    //let css_nvm = reg.capabilities.get().supports_nvm_command_set();
107    //let css_more = reg.capabilities.get().supports_more_io_command_sets();
108    // TODO: check bit 7 of css.
109
110    let config = ControllerConfig::new()
111        .with_enable(true)
112        .with_io_completion_queue_entry_size(
113            size_of::<CommonCompletion>()
114                .next_power_of_two()
115                .ilog2()
116                .try_into()
117                .unwrap(),
118        )
119        .with_io_submission_queue_entry_size(
120            size_of::<CommonCommand>()
121                .next_power_of_two()
122                .ilog2()
123                .try_into()
124                .unwrap(),
125        );
126
127    map_field!(reg.configuration).write(config);
128    while !map_field!(reg.status).read().ready() {
129        core::hint::spin_loop();
130    }
131
132    let smem = unsafe {
133        core::slice::from_raw_parts_mut(
134            saq.dma_region_mut().get_mut().as_mut_ptr() as *mut u8,
135            ADMIN_QUEUE_LEN as usize * size_of::<CommonCommand>(),
136        )
137    };
138    const C_STRIDE: usize = size_of::<CommonCompletion>();
139    const S_STRIDE: usize = size_of::<CommonCommand>();
140    let sq = nvme::queue::SubmissionQueue::new(smem, 32, S_STRIDE).unwrap();
141
142    let cmem = unsafe {
143        core::slice::from_raw_parts_mut(
144            caq.dma_region_mut().get_mut().as_mut_ptr() as *mut u8,
145            ADMIN_QUEUE_LEN as usize * size_of::<CommonCompletion>(),
146        )
147    };
148    let cq = nvme::queue::CompletionQueue::new(cmem, ADMIN_QUEUE_LEN, C_STRIDE).unwrap();
149
150    let mut saq_bell = unsafe { bar.get_mmio_offset::<u32>(0x1000) };
151    let mut caq_bell = unsafe {
152        bar.get_mmio_offset::<u32>(
153            0x1000 + 1 * map_field!(reg.capabilities).read().doorbell_stride_bytes(),
154        )
155    };
156
157    let mut admin_requester = NvmeRequester::new(
158        sq,
159        cq,
160        saq_bell.as_mut_ptr().as_raw_ptr().as_ptr(),
161        caq_bell.as_mut_ptr().as_raw_ptr().as_ptr(),
162        bar,
163        saq,
164        caq,
165    );
166
167    let cqid = DATA_QUEUE_ID.into();
168    let sqid = DATA_QUEUE_ID.into();
169
170    let req = NvmeController::create_queue_pair(
171        &mut admin_requester,
172        &mut dma_pool,
173        &mut device,
174        cqid,
175        sqid,
176        QueuePriority::Medium,
177        DATA_QUEUE_LEN as usize,
178    )?;
179
180    Ok(NvmeController {
181        inner: Arc::new(NvmeControllerInner {
182            data_requester: req,
183            admin_requester,
184            device,
185            dma_pool,
186        }),
187        capacity: OnceLock::new(),
188        block_size: OnceLock::new(),
189        int_thr: OnceLock::new(),
190    })
191}

193fn interrupt_thread_main(inner: &NvmeControllerInner, inum: usize) {
194    loop {
195        let more = inner.device.repr().check_for_interrupt(inum).is_some();
196
197        let more_a = inner.admin_requester.check_completions();
198        let more_d = inner.data_requester.check_completions();
199
200        if !more && !more_a && !more_d {
201            inner.device.repr().wait_for_interrupt(inum, None);
202        }
203    }
204}

208    pub fn new(device: Device) -> std::io::Result<Self> {
209        let dma_pool = DmaPool::new(
210            DmaPool::default_spec(),
211            twizzler_driver::dma::Access::BiDirectional,
212            DmaOptions::empty(),
213        );
214
215        let ctrl = init_controller(device, dma_pool)?;
216        let inner = ctrl.inner.clone();
217        ctrl.int_thr
218            .set(
219                std::thread::Builder::new()
220                    .name("nvme-int-0".to_string())
221                    .spawn(move || {
222                        interrupt_thread_main(&inner, 0);
223                    })
224                    .unwrap(),
225            )
226            .unwrap();
227        Ok(ctrl)
228    }

230    fn create_queue_pair(
231        admin_requester: &mut NvmeRequester,
232        dma_pool: &mut DmaPool,
233        device: &mut Device,
234        cqid: QueueId,
235        sqid: QueueId,
236        priority: QueuePriority,
237        queue_len: usize,
238    ) -> std::io::Result<NvmeRequester> {
239        let saq = dma_pool
240            .allocate_array(
241                queue_len,
242                nvme::ds::queue::subentry::CommonCommand::default(),
243            )
244            .unwrap();
245
246        let caq = dma_pool
247            .allocate_array(
248                queue_len,
249                nvme::ds::queue::comentry::CommonCompletion::default(),
250            )
251            .unwrap();
252
253        let mut saq = NvmeDmaSliceRegion::new(saq);
254        let spin = saq.dma_region_mut().pin().unwrap();
255        assert_eq!(spin.len(), 1);
256
257        let mut caq = NvmeDmaSliceRegion::new(caq);
258        let cpin = caq.dma_region_mut().pin().unwrap();
259        assert_eq!(cpin.len(), 1);
260
261        let smem = unsafe {
262            core::slice::from_raw_parts_mut(
263                saq.dma_region_mut().get_mut().as_mut_ptr() as *mut u8,
264                queue_len * size_of::<CommonCommand>(),
265            )
266        };
267
268        const C_STRIDE: usize = size_of::<CommonCompletion>();
269        const S_STRIDE: usize = size_of::<CommonCommand>();
270        let sq = nvme::queue::SubmissionQueue::new(smem, queue_len.try_into().unwrap(), S_STRIDE)
271            .unwrap();
272
273        let cmem = unsafe {
274            core::slice::from_raw_parts_mut(
275                caq.dma_region_mut().get_mut().as_mut_ptr() as *mut u8,
276                queue_len * size_of::<CommonCompletion>(),
277            )
278        };
279
280        let cq = nvme::queue::CompletionQueue::new(cmem, queue_len.try_into().unwrap(), C_STRIDE)
281            .unwrap();
282
283        {
284            // TODO: we should save these NvmeDmaRegions so they don't drop (dropping is okay, but
285            // this leaks memory )
286
287            let cmd = CreateIOCompletionQueue::new(
288                CommandId::new(),
289                cqid,
290                (&mut caq)
291                    .get_prp_list_or_buffer(PrpMode::Single, dma_pool)
292                    .unwrap(),
293                ((queue_len - 1) as u16).into(),
294                0,
295                true,
296            );
297
298            let cmd: CommonCommand = cmd.into();
299            let inflight = admin_requester.submit(cmd).unwrap();
300            loop {
301                if let Some(resp) = admin_requester.get_completion() {
302                    if resp.status().is_error() {
303                        return Err(ErrorKind::Other.into());
304                    }
305                    if inflight.id != resp.command_id().into() {
306                        tracing::error!("got other command ID for queue create command");
307                    }
308                    break;
309                }
310            }
311        }
312
313        {
314            let cmd = CreateIOSubmissionQueue::new(
315                CommandId::new(),
316                sqid,
317                (&mut saq)
318                    .get_prp_list_or_buffer(PrpMode::Single, dma_pool)
319                    .unwrap(),
320                ((queue_len - 1) as u16).into(),
321                cqid,
322                priority,
323            );
324            let cmd: CommonCommand = cmd.into();
325            let cmd: CommonCommand = cmd.into();
326            let inflight = admin_requester.submit(cmd).unwrap();
327            loop {
328                if let Some(resp) = admin_requester.get_completion() {
329                    if resp.status().is_error() {
330                        return Err(ErrorKind::Other.into());
331                    }
332                    if inflight.id != resp.command_id().into() {
333                        tracing::error!("got other command ID for queue create command");
334                    }
335                    break;
336                }
337            }
338        }
339
340        let bar = device.get_mmio(1).unwrap();
341        let reg = unsafe {
342            bar.get_mmio_offset::<nvme::ds::controller::properties::ControllerProperties>(0)
343        };
344        let reg = reg.into_ptr();
345        let bell_stride: usize = map_field!(reg.capabilities).read().doorbell_stride_bytes();
346        let mut saq_bell = unsafe {
347            bar.get_mmio_offset::<u32>(0x1000 + (u16::from(sqid) as usize) * 2 * bell_stride)
348        };
349        let mut caq_bell = unsafe {
350            bar.get_mmio_offset::<u32>(0x1000 + ((u16::from(cqid) as usize) * 2 + 1) * bell_stride)
351        };
352
353        let req = NvmeRequester::new(
354            sq,
355            cq,
356            saq_bell.as_mut_ptr().as_raw_ptr().as_ptr(),
357            caq_bell.as_mut_ptr().as_raw_ptr().as_ptr(),
358            bar,
359            saq,
360            caq,
361        );
362
363        Ok(req)
364    }

366    pub fn send_identify_controller(
367        &self,
368    ) -> Option<(

765        &self,
766        lba_start: u64,
767        in_buffer: &[u8],
768    ) -> std::io::Result<()> {
769        let nr_blocks = DMA_PAGE_SIZE * NR / self.blocking_get_lba_size();
770        let mut buffer = self
771            .inner
772            .dma_pool
773            .allocate_array(NR * DMA_PAGE_SIZE, 0u8)
774            .unwrap();
775        buffer.with_mut(0..buffer.len(), |data| data.copy_from_slice(in_buffer));
776        let mut buffer = NvmeDmaSliceRegion::new(buffer);
777        let dptr = (&mut buffer)
778            .get_dptr(
779                nvme::hosted::memory::DptrMode::Prp(PrpMode::Double),
780                &self.inner.dma_pool,
781            )
782            .unwrap();
783        // TODO: queue full
784        let inflight = self
785            .send_write_page(lba_start, dptr, nr_blocks, true)
786            .unwrap();
787
788        /*
789        let cc = loop {
790            inflight.req.get_completion();
791            if let Ok(cc) = inflight.poll() {
792                if cc.command_id() == inflight.id.into() {
793                    break cc;
794                }
795            }
796        };
797        */
798        let cc = inflight.wait()?;
799
800        if cc.status().is_error() {
