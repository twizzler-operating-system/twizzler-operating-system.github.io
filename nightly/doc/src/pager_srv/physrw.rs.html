19fn get_object(ptr: *const u8) -> (ObjID, usize) {
20    let handle = twizzler_rt_abi::object::twz_rt_get_object_handle(ptr).unwrap();
21    (handle.id(), unsafe {
22        ptr.offset_from_unsigned(handle.start())
23    })
24}

26#[derive(Clone)]
27struct Request {
28    req: RequestFromPager,
29    waiter: Arc<Waiter<CompletionToPager>>,
30}
31
32impl Request {
33    fn new(req: RequestFromPager) -> Self {
34        Self {
35            req,
36            waiter: Arc::new(Waiter::default()),
37        }
38    }
39
40    async fn wait(&self) -> CompletionToPager {
41        (&*self.waiter).await
42    }
43
44    fn finish(&self, comp: CompletionToPager) {
45        self.waiter.finish(comp);
46    }
47}

49struct PageRequestMgr {
50    _thread: JoinHandle<()>,
51    queue: QueueRef,
52    sender: Sender<Request>,
53}
54
55impl PageRequestMgr {
56    pub async fn submit_and_wait(&self, req: RequestFromPager) -> CompletionToPager {
57        let req = Request::new(req);
58        let waiter = req.clone();
59        self.sender.send(req).unwrap();
60        waiter.wait().await
61    }
62}

64static PR_MGR: OnceLock<PageRequestMgr> = OnceLock::new();
65
66fn pr_mgr_thread_main(recv: Receiver<Request>) {
67    loop {
68        match recv.recv().ok() {
69            Some(req) => {
70                let pr_mgr = PR_MGR.wait();
71                pr_mgr
72                    .queue
73                    .submit_no_wait(req.req, SubmissionFlags::empty());
74                if let Some(comp) = pr_mgr.queue.wait_for_completion() {
75                    req.finish(comp.1);
76                    unsafe { pr_mgr.queue.release_id(comp.0) };
77                }
78            }
79            None => break,
80        }
81    }
82}

84pub fn init_pr_mgr(queue: QueueRef) {
85    let (sender, recv) = mpsc::channel();
86    let _ = PR_MGR.set(PageRequestMgr {
87        _thread: std::thread::Builder::new()
88            .name("pager-requester-thread".to_owned())
89            .spawn(|| pr_mgr_thread_main(recv))
90            .unwrap(),
91        sender,
92        queue,
93    });
94}

96fn pr_mgr() -> &'static PageRequestMgr {

100pub async fn register_phys(start: u64, len: u64) -> Result<()> {
101    let request = RequestFromPager::new(PagerRequest::RegisterPhys(start, len));
102    let comp = pr_mgr().submit_and_wait(request).await;
103    match comp.data() {
104        twizzler_abi::pager::PagerCompletionData::Okay => Ok(()),
105        twizzler_abi::pager::PagerCompletionData::Error(e) => Err(e.error()),
106        _ => Err(TwzError::INVALID_ARGUMENT),
107    }
108}

110async fn do_physrw_request(
111    target_object: ObjID,
112    offset: usize,
113    len: usize,
114    phys: PhysRange,

116) -> Result<()> {

118        target_object,
119        offset,
120        len,
121        phys,
122        write_phys,

124    let comp = pr_mgr().submit_and_wait(request).await;

128        _ => Err(TwzError::INVALID_ARGUMENT),
129    }
130}

132/// Writes phys.len() bytes from the buffer into physical addresses specified in phys. If the
133/// supplied buffer is shorter than the physical range, then the remaining bytes in the physical
134/// memory are filled with 0.
135pub async fn fill_physical_pages(buf: &[u8], phys: PhysRange) -> Result<()> {
136    let obj = get_object(buf.as_ptr());
137    do_physrw_request(obj.0, obj.1, buf.len(), phys, true).await
138}
