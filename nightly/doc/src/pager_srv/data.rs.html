49impl PerObjectInner {
50    pub fn track(&mut self, obj_range: ObjectRange, phys_range: PhysRange, version: u64) {
51        assert_eq!(obj_range.len(), phys_range.len());
52        for (op, pp) in obj_range.pages().zip(phys_range.pages()) {
53            let entry = self.sync_map.entry(op).or_default();
54            if entry.version <= version {
55                entry.paddr = pp * PAGE;
56            }
57        }
58    }
59
60    fn drain_pending_syncs(&mut self) -> impl Iterator<Item = (ObjectRange, PhysRange, u64)> + '_ {
61        self.sync_map.drain().map(|(obj_page, pp)| {
62            (
63                ObjectRange::new(obj_page * PAGE, (obj_page + 1) * PAGE),
64                PhysRange::new(pp.paddr, pp.paddr + PAGE),
65                pp.version,
66            )
67        })
68    }
69
70    pub fn new(id: ObjID) -> Self {
71        Self {
72            id,
73            ..Default::default()
74        }
75    }

87impl PerObject {
88    async fn do_sync_region(
89        &self,
90        ctx: &'static PagerContext,
91        info: &ObjectEvictInfo,
92    ) -> (usize, CompletionToKernel) {
93        let pages = {
94            let mut inner = self.inner.1.lock().await;
95            inner.track(info.range, info.phys, info.version);
96            while inner.syncing {
97                self.inner.0.wait_no_relock(inner).await;
98                inner = self.inner.1.lock().await;
99            }
100            inner.syncing = true;
101            let mut pages = inner
102                .drain_pending_syncs()
103                .map(|p| (p.0, vec![PagedPhysMem::new(p.1)]))
104                .collect::<Vec<_>>();
105            pages.sort_by_key(|p| p.0);
106            let pages = pages
107                .into_iter()
108                .coalesce(|mut x, y| {
109                    if x.0.end == y.0.start {
110                        x.1.extend(y.1);
111                        Ok((ObjectRange::new(x.0.start, y.0.end), x.1))
112                    } else {
113                        Err((x, y))
114                    }
115                })
116                .collect::<Vec<_>>();
117            tracing::debug!("drained {:?}", pages);
118            pages
119        };
120        let reqs = pages
121            .into_iter()
122            .map(|p| {
123                let mut start_page = p.0.pages().next().unwrap();
124                if p.0.start == (MAX_SIZE as u64) - PAGE {
125                    start_page = 0;
126                }
127                let nr_pages = p.1.len();
128                assert_eq!(nr_pages, p.0.pages().count());
129                PageRequest::new_from_list(p.1, start_page as i64, nr_pages as u32)
130            })
131            .collect_vec();
132        let count = match page_out_many(ctx, self.id, reqs).await {
133            Err(e) => {
134                let mut inner = self.inner.1.lock().await;
135                inner.syncing = false;
136                self.inner.0.notify_all();
137                return (
138                    0,
139                    CompletionToKernel::new(
140                        KernelCompletionData::Error(e.into()),
141                        KernelCompletionFlags::DONE,
142                    ),
143                );
144            }
145            Ok(count) => count,
146        };
147
148        let mut inner = self.inner.1.lock().await;
149        inner.syncing = false;
150        self.inner.0.notify_all();
151
152        (
153            count,
154            CompletionToKernel::new(KernelCompletionData::Okay, KernelCompletionFlags::DONE),
155        )
156    }
157
158    pub async fn sync_region(
159        &self,
160        ctx: &'static PagerContext,
161        info: &ObjectEvictInfo,
162    ) -> (usize, CompletionToKernel) {
163        tracing::debug!("push pending sync: {:?}", info);
164        if info.flags.contains(ObjectEvictFlags::FENCE) {
165            self.do_sync_region(ctx, info).await
166        } else {
167            let mut inner = self.inner.1.lock().await;
168            inner.track(info.range, info.phys, info.version);
169            (
170                0,
171                CompletionToKernel::new(KernelCompletionData::Okay, KernelCompletionFlags::empty()),
172            )
173        }
174    }

251impl Future for MemoryWaiter {
252    type Output = u64;
253
254    fn poll(
255        self: std::pin::Pin<&mut Self>,
256        cx: &mut std::task::Context<'_>,
257    ) -> std::task::Poll<Self::Output> {
258        let mut inner = self.inner.lock().unwrap();
259
260        if let Some(page) = inner.get_next_available_page() {
261            std::task::Poll::Ready(page)
262        } else {
263            inner.waiters[self.pos] = Some(cx.waker().clone());
264            std::task::Poll::Pending
265        }
266    }
267}

275struct Region {
276    unused_start: u64,
277    end: u64,
278    stack: Vec<u64>,
279}
280
281#[allow(dead_code)]
282impl Region {
283    pub fn avail(&self) -> usize {
284        let unused = self.end - self.unused_start;
285        unused as usize + self.stack.len() * PAGE as usize
286    }
287
288    pub fn new(range: PhysRange) -> Self {
289        Self {
290            unused_start: range.start,
291            end: range.end,
292            stack: Vec::new(),
293        }
294    }
295
296    pub fn get_page(&mut self) -> Option<u64> {
297        self.stack.pop().or_else(|| {
298            if self.unused_start == self.end {
299                None
300            } else {
301                let next = self.unused_start;
302                self.unused_start += PAGE;
303                Some(next)
304            }
305        })
306    }
307
308    pub fn release_page(&mut self, page: u64) -> bool {
309        if self.unused_start - PAGE == page {
310            self.unused_start -= PAGE;
311        } else {
312            self.stack.push(page);
313        }
314        true
315    }
316
317    pub fn try_release_page(&mut self, page: u64) -> bool {
318        if self.unused_start - PAGE == page {
319            self.unused_start -= PAGE;
320            true
321        } else {
322            false
323        }
324    }
325}

327#[derive(Default)]
328struct Memory {
329    regions: Vec<Region>,
330}
331
332impl Memory {
333    pub fn push(&mut self, region: Region) {
334        self.regions.push(region);
335    }
336
337    pub fn get_page(&mut self) -> Option<u64> {
338        let i = 0;
339        while i < self.regions.len() {
340            if let Some(page) = self.regions[i].get_page() {
341                return Some(page);
342            }
343            self.regions.swap_remove(i);
344        }
345        None
346    }
347
348    pub fn free_page(&mut self, page: u64) {
349        for region in &mut self.regions {
350            if region.try_release_page(page) {
351                return;
352            }
353        }
354
355        for region in &mut self.regions {
356            if region.release_page(page) {
357                return;
358            }
359        }
360    }
361
362    pub fn available_memory(&self) -> usize {
363        self.regions.iter().map(|r| r.avail()).sum()
364    }
365}

417            tracing::info!(
418                "PAGER STATS: Available memory: {:10.10} KB, r {:3.3} KB/s w {:3.3} KB/s c {:2.2} (dt: {:2.2}s)",

510    /// Allocate a memory page and associate it with an object and range.
511    /// Page in the data from disk
512    /// Returns the physical range corresponding to the allocated page.
513    pub async fn fill_mem_pages_legacy(
514        &self,
515        ctx: &'static PagerContext,
516        id: ObjID,
517        obj_range: ObjectRange,
518    ) -> Result<Vec<(ObjectRange, PhysRange)>> {
519        let mut r = Vec::new();
520        for i in 0..(obj_range.pages().count() as u64) {
521            let range = ObjectRange::new(
522                obj_range.start + i * PAGE,
523                obj_range.start + i * PAGE + PAGE,
524            );
525            r.push((range, self.fill_mem_page(ctx, id, range).await?));
526        }
527        Ok(r)
528    }
529    /// Allocate a memory page and associate it with an object and range.
530    /// Page in the data from disk
531    /// Returns the physical range corresponding to the allocated page.
532    pub async fn fill_mem_page(
533        &self,
534        ctx: &'static PagerContext,
535        id: ObjID,
536        obj_range: ObjectRange,
537    ) -> Result<PhysRange> {
538        tracing::debug!(
539            "allocating memory page for ObjID {:?}, ObjectRange {:?}",
540            id,
541            obj_range
542        );
543        // TODO: remove this restriction
544        assert_eq!(obj_range.len(), 0x1000);
545        let phys_range = {
546            let page = match self.try_alloc_page() {
547                Ok(page) => page,
548                Err(mw) => {
549                    tracing::warn!("out of memory -- task waiting");
550                    mw.await
551                }
552            };
553            let phys_range = PhysRange::new(page, page + PAGE);
554            phys_range
555        };
556        page_in(ctx, id, obj_range, phys_range).await?;
557        tracing::debug!("memory page allocated successfully: {:?}", phys_range);
558
559        {
560            let mut inner = self.inner.lock().unwrap();
561            inner
562                .recent_stats
563                .read_pages(id, obj_range.len() / PAGE as usize);
564        }
565
566        return Ok(phys_range);
567    }
