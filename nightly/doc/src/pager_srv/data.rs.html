29pub struct PerObjectInner {
30    #[allow(dead_code)]
31    id: ObjID,
32    page_map: HashMap<PageNum, PerPageData>,
33    meta_page_map: HashMap<PageNum, PerPageData>,
34}

36impl PerObjectInner {
37    pub fn track(&mut self, obj_range: ObjectRange, phys_range: PhysRange) {
38        assert_eq!(obj_range.len(), PAGE as usize);
39        assert_eq!(phys_range.len(), PAGE as usize);
40
41        if let Some(old) = self.page_map.insert(
42            obj_range.start / PAGE,
43            PerPageData {
44                paddr: phys_range.start,
45            },
46        ) {
47            tracing::debug!("todo: release old: {:?}", old);
48        }
49    }
50
51    pub fn _track_meta(&mut self, obj_range: ObjectRange, phys_range: PhysRange) {
52        assert_eq!(obj_range.len(), PAGE as usize);
53        assert_eq!(phys_range.len(), PAGE as usize);
54
55        if let Some(old) = self.meta_page_map.insert(
56            obj_range.start / PAGE,
57            PerPageData {
58                paddr: phys_range.start,
59            },
60        ) {
61            tracing::debug!("todo: release old: {:?}", old);
62        }
63    }
64
65    pub fn lookup(&self, obj_range: ObjectRange) -> Option<PhysRange> {
66        let key = obj_range.start / PAGE;
67        self.page_map
68            .get(&key)
69            .map(|ppd| PhysRange::new(ppd.paddr, PAGE))
70    }
71
72    pub fn pages(&self) -> impl Iterator<Item = (ObjectRange, PerPageData)> + '_ {
73        self.page_map
74            .iter()
75            .map(|(x, y)| (ObjectRange::new(*x * PAGE, (*x + 1) * PAGE), *y))
76    }
77
78    pub fn meta_pages(&self) -> impl Iterator<Item = (ObjectRange, PerPageData)> + '_ {
79        self.meta_page_map
80            .iter()
81            .map(|(x, y)| (ObjectRange::new(*x * PAGE, (*x + 1) * PAGE), *y))
82    }
83
84    pub fn new(id: ObjID) -> Self {
85        Self {
86            id,
87            ..Default::default()
88        }
89    }

107    pub async fn sync(&self, ctx: &'static PagerContext) -> Result<()> {
108        tracing::debug!("sync: {}", self.id);
109        let (pages, mpages) = {
110            let inner = self.inner.lock().unwrap();
111            let total = inner.page_map.len() + inner.meta_page_map.len();
112            tracing::debug!("syncing {}: {} pages", self.id, total);
113            let mut pages = inner.pages().map(|p| (p.0, vec![p.1])).collect::<Vec<_>>();
114            pages.sort_by_key(|p| p.0);
115            let pages = pages
116                .into_iter()
117                .coalesce(|mut x, y| {
118                    if x.0.end == y.0.start {
119                        x.1.extend(y.1);
120                        Ok((ObjectRange::new(x.0.start, y.0.end), x.1))
121                    } else {
122                        Err((x, y))
123                    }
124                })
125                .collect::<Vec<_>>();
126            let mut mpages = inner.meta_pages().collect::<Vec<_>>();
127            mpages.sort_by_key(|p| p.0);
128            (pages, mpages)
129        };
130
131        let reqs = pages
132            .into_iter()
133            .map(|p| {
134                let start_page = p.0.pages().next().unwrap();
135                let nr_pages = p.1.len();
136                assert_eq!(nr_pages, p.0.pages().count());
137                PageRequest::new(
138                    ctx.disk
139                        .new_paging_request::<DiskPageRequest>(p.1.into_iter().map(|pd| pd.paddr)),
140                    start_page as i64,
141                    nr_pages as u32,
142                )
143            })
144            .collect_vec();
145
146        page_out_many(ctx, self.id, reqs).await?;
147
148        for p in mpages {
149            let phys_range = PhysRange::new(p.1.paddr, p.1.paddr + PAGE);
150            tracing::trace!("sync: meta page: {:?} {:?}", p, phys_range);
151            page_out(ctx, self.id, p.0, phys_range).await?;
152        }
153
154        Ok(())
155    }
156}

193struct Region {
194    unused_start: u64,
195    end: u64,
196    stack: Vec<u64>,
197}
198
199#[allow(dead_code)]
200impl Region {
201    pub fn avail(&self) -> usize {
202        let unused = self.end - self.unused_start;
203        unused as usize + self.stack.len() * PAGE as usize
204    }
205
206    pub fn new(range: PhysRange) -> Self {
207        Self {
208            unused_start: range.start,
209            end: range.end,
210            stack: Vec::new(),
211        }
212    }
213    pub fn get_page(&mut self) -> Option<u64> {
214        self.stack.pop().or_else(|| {
215            if self.unused_start == self.end {
216                None
217            } else {
218                let next = self.unused_start;
219                self.unused_start += PAGE;
220                Some(next)
221            }
222        })
223    }
224
225    pub fn release_page(&mut self, page: u64) {
226        if self.unused_start - PAGE == page {
227            self.unused_start -= PAGE;
228        } else {
229            self.stack.push(page);
230        }
231    }
232}

245        for region in &mut self.regions {

266    /// Get the next available page number and mark it as used.

303    /// Allocate a memory page and associate it with an object and range.
304    /// Page in the data from disk
305    /// Returns the physical range corresponding to the allocated page.
306    pub async fn fill_mem_page(
307        &self,
308        ctx: &'static PagerContext,
309        id: ObjID,
310        obj_range: ObjectRange,
311    ) -> Result<PhysRange> {
312        {
313            // See if we already allocated
314            let mut inner = self.inner.lock().unwrap();
315            let po = inner.get_per_object(id);
316            if let Some(phys_range) = po.lookup(obj_range) {
317                tracing::debug!(
318                    "already paged memory page for ObjID {:?}, ObjectRange {:?}",
319                    id,
320                    obj_range
321                );
322                return Ok(phys_range);
323            }
324        }
325        tracing::debug!(
326            "allocating memory page for ObjID {:?}, ObjectRange {:?}",
327            id,
328            obj_range
329        );
330        // TODO: remove this restriction
331        assert_eq!(obj_range.len(), 0x1000);
332        let phys_range = {
333            let mut inner = self.inner.lock().unwrap();
334            let page = inner.get_mem_page();
335            let phys_range = PhysRange::new(page, page + PAGE);
336            let po = inner.get_per_object_mut(id);
337            po.track(obj_range, phys_range);
338            phys_range
339        };
340        page_in(ctx, id, obj_range, phys_range).await?;
341        tracing::debug!("memory page allocated successfully: {:?}", phys_range);
342        return Ok(phys_range);
343    }

345    pub fn lookup_object(&self, ctx: &PagerContext, id: ObjID) -> Option<ObjectInfo> {
346        let mut b = [];
347        if objid_to_ino(id.raw()).is_some() {
348            ctx.paged_ostore.find_external(id.raw()).ok()?;
349            return Some(ObjectInfo::new(id));
350        }
351        ctx.paged_ostore.read_object(id.raw(), 0, &mut b).ok()?;
352        Some(ObjectInfo::new(id))
353    }

355    pub async fn sync(&self, ctx: &'static PagerContext, id: ObjID) {
356        let po = {
357            let mut inner = self.inner.lock().unwrap();
358            inner.get_per_object(id).clone()
359        };
360        let _ = po.sync(ctx).await.inspect_err(|e| {
361            tracing::warn!("sync failed for {}: {}", id, e);
362        });
363        ctx.paged_ostore.flush().unwrap();
364    }
