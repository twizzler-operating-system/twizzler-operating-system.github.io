24pub(crate) enum RelocState {

27    /// Relocation has started, but not finished, or failed.

146    pub fn is_binary(&self) -> bool {
147        let Some(dynamic) = self
148            .get_elf()
149            .ok()
150            .and_then(|elf| elf.dynamic().ok())
151            .flatten()
152        else {
153            return false;
154        };
155        let Some(flags) = dynamic.iter().find_map(|ent| {
156            if ent.d_tag == DT_FLAGS_1 {
157                Some(ent.d_val())
158            } else {
159                None
160            }
161        }) else {
162            return false;
163        };
164        flags & elf::abi::DF_1_PIE as u64 != 0
165    }

177    /// Get a raw pointer to the program headers for this library.
178    pub fn get_phdrs_raw(&self) -> Option<(*const Elf64_Phdr, usize)> {
179        Some((
180            self.get_elf().ok()?.segments()?.iter().find_map(|p| {
181                if p.p_type == PT_PHDR {
182                    Some(self.laddr(p.p_vaddr))
183                } else {
184                    None
185                }
186            })?,
187            self.get_elf().ok()?.segments()?.len(),
188        ))
189    }

191    /// Return a handle to the full ELF file.

193        elf::ElfBytes::minimal_parse(self.full_obj.slice())
194    }

196    /// Get the load address for this library.

201    /// Compute an in-memory address for a ELF virtual addr.
202    pub fn laddr<T>(&self, val: u64) -> *const T {
203        (self.base_addr() + val as usize) as *const T
204    }
205
206    /// Compute an in-memory address (mut) for a ELF virtual addr.
207    pub fn laddr_mut<T>(&self, val: u64) -> *mut T {
208        (self.base_addr() + val as usize) as *mut T
209    }

211    /// Get a function pointer to this library's entry address, if one exists.
212    pub fn get_entry_address(
213        &self,
214    ) -> Result<extern "C" fn(*const RuntimeInfo) -> !, DynlinkError> {
215        let entry = self.get_elf()?.ehdr.e_entry;
216        if entry == 0 {
217            return Err(DynlinkErrorKind::NoEntryAddress {
218                name: self.name.clone(),
219            }
220            .into());
221        }
222        let entry: *const u8 = self.laddr(entry);
223        let ptr: extern "C" fn(*const RuntimeInfo) -> ! =
224            unsafe { core::mem::transmute(entry as usize) };
225        Ok(ptr)
226    }

228    // Helper to find the TLS program header.
229    fn get_tls_phdr(&self) -> Result<Option<ProgramHeader>, DynlinkError> {
230        Ok(self
231            .get_elf()?
232            .segments()
233            .and_then(|phdrs| phdrs.iter().find(|phdr| phdr.p_type == PT_TLS)))
234    }
235
236    pub(crate) fn get_tls_data(&self) -> Result<Option<&[u8]>, DynlinkError> {
237        let phdr = self.get_tls_phdr()?;
238        Ok(phdr.map(|phdr| {
239            let slice = unsafe {
240                core::slice::from_raw_parts(self.laddr(phdr.p_vaddr), phdr.p_memsz as usize)
241            };
242            slice
243        }))
244    }

246    fn do_lookup_symbol(
247        &self,
248        name: &str,
249        allow_weak: bool,
250    ) -> Result<RelocatedSymbol<'_>, DynlinkError> {
251        let elf = self.get_elf()?;
252        let common = elf.find_common_data()?;
253        tracing::trace!("lookup {} in {}", name, self.name);
254
255        /*
256        if self.is_relocated() {
257            if let Some(gates) = self.iter_secgates() {
258                for sc in gates {
259                    if let Ok(gname) = sc.name().to_str() {
260                        if gname == name {
261                            tracing::info!("found as secure gate");
262                            return Ok(RelocatedSymbol::new_sc(sc.imp, self));
263                        }
264                    }
265                }
266            }
267        }
268        */
269
270        // Try the GNU hash table, if present.
271        if let Some(h) = &common.gnu_hash {
272            if let Some((_, sym)) = h
273                .find(
274                    name.as_ref(),
275                    common
276                        .dynsyms
277                        .as_ref()
278                        .ok_or_else(|| DynlinkErrorKind::MissingSection {
279                            name: "dynsyms".to_string(),
280                        })?,
281                    common.dynsyms_strs.as_ref().ok_or_else(|| {
282                        DynlinkErrorKind::MissingSection {
283                            name: "dynsyms_strs".to_string(),
284                        }
285                    })?,
286                )
287                .ok()
288                .flatten()
289            {
290                if !sym.is_undefined() {
291                    // TODO: proper weak symbol handling.
292                    if sym.st_bind() != STB_WEAK
293                        || allow_weak
294                        || (self.is_relocated() && self.is_secgate(name))
295                    {
296                        return Ok(RelocatedSymbol::new(sym, self));
297                    } else {
298                        tracing::debug!("lookup symbol {} skipping weak binding in {}", name, self);
299                    }
300                } else {
301                    tracing::info!("undefined symbol: {}", name);
302                }
303            }
304            return Err(DynlinkErrorKind::NameNotFound {
305                name: name.to_string(),
306            }
307            .into());
308        }
309
310        // Try the sysv hash table, if present.
311        if let Some(h) = &common.sysv_hash {
312            if let Some((_, sym)) = h
313                .find(
314                    name.as_ref(),
315                    common
316                        .dynsyms
317                        .as_ref()
318                        .ok_or_else(|| DynlinkErrorKind::MissingSection {
319                            name: "dynsyms".to_string(),
320                        })?,
321                    common.dynsyms_strs.as_ref().ok_or_else(|| {
322                        DynlinkErrorKind::MissingSection {
323                            name: "dynsyms_strs".to_string(),
324                        }
325                    })?,
326                )
327                .ok()
328                .flatten()
329            {
330                if !sym.is_undefined() {
331                    // TODO: proper weak symbol handling.
332                    if sym.st_bind() != STB_WEAK
333                        || allow_weak
334                        || (self.is_relocated() && self.is_secgate(name))
335                    {
336                        return Ok(RelocatedSymbol::new(sym, self));
337                    } else {
338                        tracing::info!("lookup symbol {} skipping weak binding in {}", name, self);
339                    }
340                } else {
341                    tracing::info!("undefined symbol: {}", name);
342                }
343            }
344        }
345
346        Err(DynlinkErrorKind::NameNotFound {
347            name: name.to_string(),
348        }
349        .into())
350    }

352    pub(crate) fn lookup_symbol(
353        &self,
354        name: &str,
355        allow_weak: bool,
356        allow_prefix: bool,
357    ) -> Result<RelocatedSymbol<'_>, DynlinkError> {
358        let ret = self.do_lookup_symbol(&name, allow_weak);
359        if allow_prefix && ret.is_err() && !name.starts_with("__TWIZZLER_SECURE_GATE_") {
360            let name = format!("__TWIZZLER_SECURE_GATE_{}", name);
361            if let Ok(o) = self.do_lookup_symbol(&name, allow_weak) {
362                return Ok(o);
363            }
364        }
365        ret
366    }

368    pub(crate) fn is_local_or_secgate_from(&self, other: &Library, name: &str) -> bool {

376    pub fn is_relocated(&self) -> bool {
377        self.reloc_state == RelocState::Relocated
378    }

380    fn is_secgate(&self, name: &str) -> bool {
381        self.iter_secgates()
382            .map(|gates| {
383                gates
384                    .iter()
385                    .any(|gate| gate.name().to_bytes() == name.as_bytes())
386            })
387            .unwrap_or(false)
388    }

390    pub fn iter_secgates(&self) -> Option<&[RawSecGateInfo]> {
391        let addr = self.secgate_info.info_addr?;
392
393        Some(unsafe { core::slice::from_raw_parts(addr as *const _, self.secgate_info.num) })
394    }
