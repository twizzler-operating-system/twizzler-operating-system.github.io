29/// Tracks info for loaded, but not yet running, compartments.
30#[derive(Debug)]
31pub struct RunCompLoader {
32    loaded_extras: Vec<LoadInfo>,
33    root_comp: LoadInfo,
34}
35
36/// A single compartment, loaded but not yet running.
37#[derive(Debug, Clone)]
38struct LoadInfo {
39    // root (eg executable) library ID
40    #[allow(dead_code)]
41    root_id: LibraryId,
42    // runtime library ID (maybe injected)
43    #[allow(dead_code)]
44    rt_id: LibraryId,
45    // security context ID
46    sctx_id: ObjID,
47    name: String,
48    comp_id: CompartmentId,
49    // all constructors for all libraries
50    ctor_info: Vec<CtorSet>,
51    // entry point to call for the runtime to init this compartment
52    entry: extern "C" fn(*const RuntimeInfo) -> !,
53    is_binary: bool,
54}
55
56impl LoadInfo {
57    fn new(
58        dynlink: &mut Context,
59        root_id: LibraryId,
60        rt_id: LibraryId,
61        sctx_id: ObjID,
62        is_binary: bool,
63        extras: &[LibraryId],
64    ) -> Result<Self, DynlinkError> {
65        let lib = dynlink.get_library(rt_id)?;
66        let extra_ctors: Vec<_> = extras
67            .iter()
68            .map(|extra| dynlink.build_ctors_list(*extra, Some(lib.compartment())))
69            .try_collect()?;
70        let mut root_ctors = dynlink.build_ctors_list(root_id, Some(lib.compartment()))?;
71        let mut ctor_info: Vec<_> = extra_ctors.iter().flatten().copied().collect();
72        ctor_info.append(&mut root_ctors);
73        Ok(Self {
74            root_id,
75            rt_id,
76            comp_id: lib.compartment(),
77            sctx_id,
78            name: dynlink.get_compartment(lib.compartment())?.name.clone(),
79            ctor_info,
80            entry: lib.get_entry_address()?,
81            is_binary,
82        })
83    }
84
85    fn build_runcomp(
86        &self,
87        handle: MapHandle,
88        stack_object: StackObject,
89    ) -> Result<RunComp, DynlinkError> {
90        let comp_config =
91            CompConfigObject::new(handle, SharedCompConfig::new(self.sctx_id, null_mut()));
92
93        let flags = if self.is_binary { COMP_IS_BINARY } else { 0 };
94        Ok(RunComp::new(
95            self.sctx_id,
96            self.sctx_id,
97            self.name.clone(),
98            self.comp_id,
99            vec![],
100            comp_config,
101            flags,
102            stack_object,
103            self.entry as usize,
104            &self.ctor_info,
105        ))
106    }
107}

115const RUNTIME_NAME: &str = "libtwz_rt.so";
116
117impl RunCompLoader {
118    // the runtime library might be in the dependency tree from the shared object files.
119    // if not, we need to insert it.
120    fn maybe_inject_runtime(
121        dynlink: &mut Context,
122        root_id: LibraryId,
123        comp_id: CompartmentId,
124        load_ctx: &mut LoadCtx,
125    ) -> Result<LibraryId, DynlinkError> {
126        if let Some(id) = dynlink.lookup_library(comp_id, RUNTIME_NAME) {
127            return Ok(id);
128        }
129
130        let rt_unlib = UnloadedLibrary::new(RUNTIME_NAME);
131        let loads = dynlink.load_library_in_compartment(
132            comp_id,
133            rt_unlib,
134            AllowedGates::Private,
135            load_ctx,
136        )?;
137        dynlink.add_manual_dependency(root_id, loads[0].lib);
138        Ok(loads[0].lib)
139    }
140
141    /// Build a new RunCompLoader. This will load and relocate libraries in the dynamic linker, but
142    /// won't start compartment threads.
143    pub fn new(
144        dynlink: &mut Context,
145        comp_name: &str,
146        root_unlib: UnloadedLibrary,
147        extras: &[UnloadedLibrary],
148        new_comp_flags: NewCompartmentFlags,
149        mondebug: bool,
150    ) -> miette::Result<Self> {
151        struct UnloadOnDrop(Vec<LoadIds>);
152        impl Drop for UnloadOnDrop {
153            fn drop(&mut self) {
154                tracing::warn!("todo: drop");
155            }
156        }
157        let root_comp_id = dynlink.add_compartment(comp_name, new_comp_flags)?;
158        let allowed_gates = if new_comp_flags.contains(NewCompartmentFlags::EXPORT_GATES) {
159            AllowedGates::Public
160        } else {
161            AllowedGates::Private
162        };
163        let mut load_ctx = LoadCtx::default();
164
165        let extra_load_ids: Vec<_> = extras
166            .into_iter()
167            .map(|extra| {
168                if mondebug {
169                    tracing::info!("loading ld preload library: {}", extra.name);
170                } else {
171                    tracing::debug!("loading ld preload library: {}", extra.name);
172                }
173                dynlink.load_library_in_compartment(
174                    root_comp_id,
175                    extra.clone(),
176                    AllowedGates::Private,
177                    &mut load_ctx,
178                )
179            })
180            .try_collect()?;
181
182        let mut loads = UnloadOnDrop(dynlink.load_library_in_compartment(
183            root_comp_id,
184            root_unlib.clone(),
185            allowed_gates,
186            &mut load_ctx,
187        )?);
188
189        for extra in &extra_load_ids {
190            for extra in extra {
191                loads.0.push(extra.clone());
192            }
193        }
194
195        // The dynamic linker gives us a list of loaded libraries, and which compartments they ended
196        // up in. For each of those, we may need to inject the runtime library. Collect all
197        // the information about the extra compartments.
198        let mut cache = HashSet::new();
199        let extra_compartments = loads.0.iter().filter_map(|load| {
200            if load.comp != root_comp_id {
201                // This compartment was loaded in addition to the root comp as part of our
202                // initial load request. Check if we haven't seen it before.
203                if cache.contains(&load.comp) {
204                    return None;
205                }
206                cache.insert(load.comp);
207
208                // Inject the runtime library, careful to collect the error and keep going.
209                let rt_id =
210                    match Self::maybe_inject_runtime(dynlink, load.lib, load.comp, &mut load_ctx) {
211                        Ok(id) => id,
212                        Err(e) => return Some(Err(e)),
213                    };
214                Some(LoadInfo::new(
215                    dynlink,
216                    load.lib,
217                    rt_id,
218                    *load_ctx.set.get(&load.comp).unwrap(),
219                    false,
220                    &[],
221                ))
222            } else {
223                None
224            }
225        });
226
227        let extra_compartments = DynlinkError::collect(
228            dynlink::DynlinkErrorKind::CompartmentLoadFail {
229                compartment: comp_name.to_string(),
230            },
231            extra_compartments,
232        )?;
233
234        let root_id = loads.0[0].lib;
235        let rt_id = Self::maybe_inject_runtime(dynlink, root_id, root_comp_id, &mut load_ctx)?;
236        let extra_lids = extra_load_ids
237            .iter()
238            .flatten()
239            .map(|x| x.lib)
240            .collect::<Vec<_>>();
241        for extra in &extra_lids {
242            dynlink.relocate_all(*extra)?;
243        }
244        dynlink.relocate_all(root_id)?;
245
246        let is_binary = dynlink.get_library(root_id)?.is_binary();
247        let root_comp = LoadInfo::new(
248            dynlink,
249            root_id,
250            rt_id,
251            *load_ctx.set.get(&root_comp_id).unwrap(),
252            is_binary,
253            extra_lids.as_slice(),
254        )?;
255
256        if mondebug {
257            let print_comp = |cmp: &LoadInfo| -> miette::Result<()> {
258                let dcmp = dynlink.get_compartment(cmp.comp_id)?;
259                tracing::info!("Loaded libraries for {}:", &dcmp.name);
260                for lid in dcmp.library_ids() {
261                    let lib = dynlink.get_library(lid)?;
262                    let mut flags = ["-", "-", "-"];
263                    if lib.is_binary() {
264                        flags[0] = "B";
265                    } else {
266                        flags[0] = "l";
267                    }
268                    if lib.id() == cmp.rt_id {
269                        flags[1] = "r";
270                    } else if lib.id() == cmp.root_id {
271                        flags[1] = "R";
272                    }
273                    if lib.allows_gates() {
274                        flags[2] = "g";
275                    }
276                    let flags = flags.join("");
277                    tracing::info!("{:16x} {} {}", lib.base_addr(), flags, &lib.name);
278                    if let Some(isg) = lib.iter_secgates() {
279                        for gate in isg {
280                            tracing::info!(
281                                "    GATE {:16x} {}",
282                                gate.imp,
283                                gate.name().to_string_lossy()
284                            )
285                        }
286                    }
287                }
288                Ok(())
289            };
290            tracing::info!("Load info for {}", comp_name);
291            let _ = print_comp(&root_comp);
292            for cmp in &extra_compartments {
293                let _ = print_comp(cmp);
294            }
295        }
296
297        // We don't want to drop anymore, since now drop-cleanup will be handled by RunCompLoader.
298        std::mem::forget(loads);
299        Ok(RunCompLoader {
300            loaded_extras: extra_compartments,
301            root_comp,
302        })
303    }

305    pub fn build_rcs(
306        self,
307        cmp: &mut CompartmentMgr,
308        dynlink: &mut Context,
309        _mondebug: bool,
310    ) -> miette::Result<ObjID> {
311        let make_new_handle = |id| {
312            Space::safe_create_and_map_runtime_object(
313                &get_monitor().space,
314                id,
315                MapFlags::READ | MapFlags::WRITE,
316            )
317        };
318
319        let root_rc = self.root_comp.build_runcomp(
320            make_new_handle(self.root_comp.sctx_id)?,
321            StackObject::new(make_new_handle(self.root_comp.sctx_id)?, DEFAULT_STACK_SIZE)?,
322        )?;
323
324        let mut ids = vec![root_rc.instance];
325        // Make all the handles first, for easier cleanup.
326        let handles = self
327            .loaded_extras
328            .iter()
329            .map(|extra| {
330                Ok::<_, miette::Report>((
331                    make_new_handle(extra.sctx_id)?,
332                    StackObject::new(make_new_handle(extra.sctx_id)?, DEFAULT_STACK_SIZE)?,
333                ))
334            })
335            .try_collect::<Vec<_>>()?;
336        // Construct the RunComps for all the extra compartments.
337        let mut extras = self
338            .loaded_extras
339            .iter()
340            .zip(handles)
341            .map(|extra| extra.0.build_runcomp(extra.1 .0, extra.1 .1))
342            .try_collect::<Vec<_>>()?;
343
344        for rc in extras.drain(..) {
345            ids.push(rc.instance);
346            cmp.insert(rc);
347        }
348        cmp.insert(root_rc);
349        std::mem::forget(self);
350
351        // Set all the dependency information.
352        for id in &ids {
353            let Ok(comp) = cmp.get(*id) else { continue };
354            let mut deps = dynlink
355                .compartment_dependencies(comp.compartment_id)?
356                .iter()
357                .filter_map(|item| cmp.get_dynlinkid(*item).map(|rc| rc.instance).ok())
358                .collect();
359            cmp.get_mut(*id).unwrap().deps.append(&mut deps);
360
361            let Ok(comp) = cmp.get(*id) else { continue };
362            tracing::debug!("set comp {} deps to {:?}", comp.name, comp.deps);
363        }
364        Self::rec_inc_all_use_counts(cmp, ids[0], &HashSet::from_iter(ids.iter().cloned()));
365
366        Ok(ids[0])
367    }
368
369    fn rec_inc_all_use_counts(
370        cmgr: &mut CompartmentMgr,
371        start: ObjID,
372        created: &HashSet<ObjID>,
373    ) -> Option<()> {
374        debug_assert!(created.contains(&start));
375        let rc = cmgr.get(start).ok()?;
376        for dep in rc.deps.clone() {
377            if created.contains(&dep) {
378                Self::rec_inc_all_use_counts(cmgr, dep, created);
379            }
380            if let Ok(rc) = cmgr.get_mut(dep) {
381                rc.inc_use_count();
382            }
383        }
384
385        Some(())
386    }

389impl Monitor {
390    pub(crate) fn start_compartment(
391        &self,
392        instance: ObjID,
393        args: &[&CStr],
394        env: &[&CStr],
395        mondebug: bool,
396    ) -> Result<(), TwzError> {
397        if mondebug {
398            tracing::info!("start compartment {}: {:?} {:?}", instance, args, env);
399        }
400        let deps = {
401            let cmp = self.comp_mgr.read(ThreadKey::get().unwrap());
402            let rc = cmp.get(instance)?;
403            tracing::debug!(
404                "starting compartment {} ({}) (binary = {})",
405                rc.name,
406                rc.instance,
407                rc.has_flag(COMP_IS_BINARY)
408            );
409            rc.deps.clone()
410        };
411        for dep in deps {
412            self.start_compartment(dep, &[], env, false)?;
413        }
414        // Check the state of this compartment.
415        let state = self.load_compartment_flags(instance);
416        if state & COMP_EXITED != 0 || state & COMP_DESTRUCTED != 0 {
417            tracing::error!(
418                "tried to start compartment ({:?}, {}) that has already exited (state: {:x})",
419                self.comp_name(instance),
420                instance,
421                state
422            );
423            return Err(GenericError::Internal.into());
424        }
425
426        loop {
427            // Check the state of this compartment.
428            let state = self.load_compartment_flags(instance);
429            if state & COMP_READY != 0 {
430                return Ok(());
431            }
432            let info = {
433                let (ref mut tmgr, ref mut cmp, ref mut dynlink, _, _) =
434                    *self.locks.lock(ThreadKey::get().unwrap());
435                let rc = cmp.get_mut(instance)?;
436
437                rc.start_main_thread(state, &mut *tmgr, &mut *dynlink, args, env)
438            };
439            if info.is_none() {
440                return Err(GenericError::Internal.into());
441            }
442            self.wait_for_compartment_state_change(instance, state);
443        }
444    }
