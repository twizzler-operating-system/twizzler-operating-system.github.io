30/// Compartment is ready (loaded, reloacated, runtime started and ctors run).
31pub const COMP_READY: u64 = CompartmentFlags::READY.bits();
32/// Compartment is a binary, not a library.
33pub const COMP_IS_BINARY: u64 = CompartmentFlags::IS_BINARY.bits();
34/// Compartment runtime thread may exit.
35pub const COMP_THREAD_CAN_EXIT: u64 = CompartmentFlags::THREAD_CAN_EXIT.bits();
36/// Compartment thread has been started once.
37pub const COMP_STARTED: u64 = CompartmentFlags::STARTED.bits();
38/// Compartment destructors have run.
39pub const COMP_DESTRUCTED: u64 = CompartmentFlags::DESTRUCTED.bits();
40/// Compartment thread has exited.
41pub const COMP_EXITED: u64 = CompartmentFlags::EXITED.bits();

88/// Per-thread data in a compartment.
89pub struct PerThread {
90    simple_buffer: Option<(SimpleBuffer, MapHandle)>,
91}
92
93impl PerThread {
94    /// Create a new PerThread. Note that this must succeed, so any allocation failures must be
95    /// handled gracefully. This means that if the thread fails to allocate a simple buffer, it
96    /// will just forego having one. This may cause a failure down the line, but it's the best we
97    /// can do without panicing.
98    fn new(instance: ObjID, _th: ObjID) -> Self {
99        let handle = Space::safe_create_and_map_runtime_object(
100            &get_monitor().space,
101            instance,
102            MapFlags::READ | MapFlags::WRITE,
103        )
104        .ok();
105
106        Self {
107            simple_buffer: handle
108                .map(|handle| (SimpleBuffer::new(unsafe { handle.object_handle() }), handle)),
109        }
110    }
111
112    /// Write bytes into this compartment-thread's simple buffer.
113    pub fn write_bytes(&mut self, bytes: &[u8]) -> usize {
114        self.simple_buffer
115            .as_mut()
116            .map(|sb| sb.0.write(bytes))
117            .unwrap_or(0)
118    }
119
120    /// Read bytes from this compartment-thread's simple buffer.
121    pub fn read_bytes(&mut self, len: usize) -> Vec<u8> {
122        let mut v = vec![0; len];
123        let readlen = self
124            .simple_buffer
125            .as_mut()
126            .map(|sb| sb.0.read(&mut v))
127            .unwrap_or(0);
128        v.truncate(readlen);
129        v
130    }
131
132    /// Get the Object ID of this compartment thread's simple buffer.
133    pub fn simple_buffer_id(&self) -> Option<ObjID> {
134        Some(self.simple_buffer.as_ref()?.0.handle().id())
135    }
136}

155        Self {

168            use_count: 0,
169        }

209    /// Allocate some space in the compartment allocator, and initialize it.
210    pub fn monitor_new<T: Copy + Sized>(&mut self, data: T) -> Result<*mut T, ()> {
211        unsafe {
212            let place: NonNull<T> = self.alloc.malloc(Layout::new::<T>())?.cast();
213            place.as_ptr().write(data);
214            Ok(place.as_ptr())
215        }
216    }
217
218    /// Allocate some space in the compartment allocator for a slice, and initialize it.
219    pub fn monitor_new_slice<T: Copy + Sized>(&mut self, data: &[T]) -> Result<*mut T, ()> {
220        unsafe {
221            let place = self.alloc.malloc(Layout::array::<T>(data.len()).unwrap())?;
222            let slice = core::slice::from_raw_parts_mut(place.as_ptr() as *mut T, data.len());
223            slice.copy_from_slice(data);
224            Ok(place.as_ptr() as *mut T)
225        }
226    }

228    /// Set a flag on this compartment, and wakeup anyone waiting on flag change.
229    pub fn set_flag(&self, val: u64) {
230        tracing::trace!("compartment {} set flag {:x}", self.name, val);
231        self.flags.fetch_or(val, Ordering::SeqCst);
232        self.notify_state_changed();
233    }
234
235    /// Set a flag on this compartment, and wakeup anyone waiting on flag change.
236    pub fn cas_flag(&self, old: u64, new: u64) -> Result<u64, u64> {
237        let r = self
238            .flags
239            .compare_exchange(old, new, Ordering::SeqCst, Ordering::SeqCst);
240        if r.is_ok() {
241            tracing::trace!("compartment {} cas flag {:x} -> {:x}", self.name, old, new);
242            self.notify_state_changed();
243        }
244        r
245    }

247    pub fn notify_state_changed(&self) {
248        let _ = twizzler_abi::syscall::sys_thread_sync(
249            &mut [ThreadSync::new_wake(ThreadSyncWake::new(
250                ThreadSyncReference::Virtual(&*self.flags),
251                usize::MAX,
252            ))],
253            None,
254        );
255    }
256
257    /// Check if a flag is set.
258    pub fn has_flag(&self, flag: u64) -> bool {
259        self.flags.load(Ordering::SeqCst) & flag != 0
260    }
261
262    /// Setup a [ThreadSyncSleep] for waiting until the flag is set. Returns None if the flag is
263    /// already set.
264    pub fn until_change(&self, cur: u64) -> ThreadSyncSleep {
265        ThreadSyncSleep::new(
266            ThreadSyncReference::Virtual(&*self.flags),
267            cur,
268            ThreadSyncOp::Equal,
269            ThreadSyncFlags::empty(),
270        )
271    }

273    /// Get the raw flags bits for this RC.
274    pub fn raw_flags(&self) -> u64 {
275        self.flags.load(Ordering::SeqCst)
276    }

278    pub(crate) fn start_main_thread(
279        &mut self,
280        state: u64,
281        tmgr: &mut ThreadMgr,
282        dynlink: &mut Context,
283        args: &[&CStr],
284        env: &[&CStr],
285    ) -> Option<bool> {
286        if self.has_flag(COMP_STARTED) {
287            return Some(false);
288        }
289        let state = state & !COMP_STARTED;
290        if self
291            .flags
292            .compare_exchange(
293                state,
294                state | COMP_STARTED,
295                Ordering::SeqCst,
296                Ordering::SeqCst,
297            )
298            .is_err()
299        {
300            return None;
301        }
302
303        tracing::debug!("starting main thread for compartment {}", self.name);
304        debug_assert!(self.main.is_none());
305        // Unwrap-Ok: we only take this once, when starting the main thread.
306        let (stack, entry, ctors) = self.init_info.take().unwrap();
307        let mut build_init_info = || -> Option<_> {
308            let comp_config_info =
309                self.comp_config_object.get_comp_config() as *mut SharedCompConfig;
310            let ctors_in_comp = self.monitor_new_slice(&ctors).ok()?;
311
312            // TODO: unwrap
313            let mut args_in_comp: Vec<_> = args
314                .iter()
315                .map(|arg| self.monitor_new_slice(arg.to_bytes_with_nul()).unwrap())
316                .collect();
317
318            if args_in_comp.len() == 0 {
319                let cname = CString::new(self.name.as_bytes()).unwrap();
320                args_in_comp = vec![self.monitor_new_slice(cname.as_bytes()).unwrap()];
321            }
322            let argc = args_in_comp.len();
323
324            let mut envs_in_comp: Vec<_> = env
325                .iter()
326                .map(|arg| self.monitor_new_slice(arg.to_bytes_with_nul()).unwrap())
327                .collect();
328
329            args_in_comp.push(core::ptr::null_mut());
330            envs_in_comp.push(core::ptr::null_mut());
331
332            let args_in_comp_in_comp = self.monitor_new_slice(&args_in_comp).unwrap();
333            let envs_in_comp_in_comp = self.monitor_new_slice(&envs_in_comp).unwrap();
334
335            let comp_init_info = CompartmentInitInfo {
336                ctor_set_array: ctors_in_comp,
337                ctor_set_len: ctors.len(),
338                comp_config_info: comp_config_info.cast(),
339            };
340            let comp_init_info_in_comp = self.monitor_new(comp_init_info).ok()?;
341            // TODO: fill out argc and argv and envp
342            let rtinfo = RuntimeInfo {
343                flags: 0,
344                kind: RUNTIME_INIT_COMP,
345                args: args_in_comp_in_comp.cast(),
346                argc,
347                envp: envs_in_comp_in_comp.cast(),
348                init_info: InitInfoPtrs {
349                    comp: comp_init_info_in_comp,
350                },
351            };
352            self.monitor_new(rtinfo).ok()
353        };
354        let arg = match build_init_info() {
355            Some(arg) => arg as usize,
356            None => {
357                self.set_flag(COMP_EXITED);
358                return None;
359            }
360        };
361        if self.build_tls_template(dynlink).is_none() {
362            self.set_flag(COMP_EXITED);
363            return None;
364        }
365
366        let mt = match CompThread::new(
367            tmgr,
368            dynlink,
369            stack,
370            self.instance,
371            Some(self.instance),
372            entry,
373            arg,
374        ) {
375            Ok(mt) => mt,
376            Err(_) => {
377                self.set_flag(COMP_EXITED);
378                return None;
379            }
380        };
381        self.main = Some(mt);
382        self.notify_state_changed();
383
384        Some(true)
385    }

387    fn build_tls_template(&mut self, dynlink: &mut Context) -> Option<()> {
388        let region = dynlink
389            .get_compartment_mut(self.compartment_id)
390            .unwrap()
391            .build_tls_region(RuntimeThreadControl::default(), |layout| {
392                unsafe { self.alloc.malloc(layout) }.ok()
393            })
394            .ok()?;
395
396        let template: TlsTemplateInfo = region.into();
397        let tls_template = self.monitor_new(template).ok()?;
398
399        let config = self.comp_config_object.read_comp_config();
400        config.set_tls_template(tls_template);
401        self.comp_config_object.write_config(config);
402        Some(())
403    }

422    pub(crate) fn dec_use_count(&mut self) -> bool {
423        debug_assert!(self.use_count > 0);
424        self.use_count -= 1;
425
426        tracing::trace!(
427            "compartment {} dec use count -> {}",
428            self.name,
429            self.use_count
430        );
431        let z = self.use_count == 0;
432        if z {
433            self.set_flag(COMP_THREAD_CAN_EXIT);
434        }
435        z
436    }
