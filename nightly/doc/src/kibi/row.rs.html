5//! # Row
6//!
7//! Utilities for rows. A `Row` owns the underlying characters, the rendered
8//! string and the syntax highlighting information.
9
10use std::{iter::repeat_n, num::NonZeroUsize};
11
12use unicode_width::UnicodeWidthChar;
13
14use crate::ansi_escape::{RESET, WBG, push_colored};
15use crate::syntax::{Conf as SyntaxConf, HlType};
16
17/// The "Highlight State" of the row
18#[derive(Clone, Copy, Default, PartialEq, Eq)]
19pub enum HlState {
20    /// Normal state.
21    #[default]
22    Normal,
23    /// A multi-line comment has been open, but not yet closed.
24    MultiLineComment,
25    /// A string has been open with the given quote character (for instance
26    /// b'\'' or b'"'), but not yet closed.
27    String(u8),
28    /// A multi-line string has been open, but not yet closed.
29    MultiLineString,
30}
31
32/// Represents a row of characters and how it is rendered.
33#[derive(Default)]
34pub struct Row {
35    /// The characters of the row.
36    pub chars: Vec<u8>,
37    /// How the characters are rendered. In particular, tabs are converted into
38    /// several spaces, and bytes may be combined into single UTF-8
39    /// characters.
40    render: String,
41    /// Mapping from indices in `self.chars` to the corresponding indices in
42    /// `self.render`.
43    pub cx2rx: Vec<usize>,
44    /// Mapping from indices in `self.render` to the corresponding indices in
45    /// `self.chars`.
46    pub rx2cx: Vec<usize>,
47    /// The vector of `HLType` for each rendered character.
48    pub hl: Vec<HlType>,
49    /// The final state of the row.
50    pub hl_state: HlState,
51    /// If not `None`, the range that is currently matched during a FIND
52    /// operation.
53    pub match_segment: Option<std::ops::Range<usize>>,
54}
55
56impl Row {
57    /// Create a new row, containing characters `chars`.
58    pub fn new(chars: Vec<u8>) -> Self { Self { chars, cx2rx: vec![0], ..Self::default() } }
59
60    // TODO: Combine update and update_syntax
61    /// Update the row: convert tabs into spaces and compute highlight symbols
62    /// The `hl_state` argument is the `HLState` for the previous row.
63    pub fn update(&mut self, syntax: &SyntaxConf, hl_state: HlState, tab: NonZeroUsize) -> HlState {
64        let (..) = (self.render.clear(), self.cx2rx.clear(), self.rx2cx.clear());
65        let (mut cx, mut rx) = (0, 0);
66        for c in String::from_utf8_lossy(&self.chars).chars() {
67            // The number of rendered characters
68            let n_rend_chars =
69                if c == '\t' { tab.get() - (rx % tab) } else { c.width().unwrap_or(1) };
70            self.render.push_str(&(if c == '\t' { " ".repeat(n_rend_chars) } else { c.into() }));
71            self.cx2rx.extend(repeat_n(rx, c.len_utf8()));
72            self.rx2cx.extend(repeat_n(cx, n_rend_chars));
73            (rx, cx) = (rx + n_rend_chars, cx + c.len_utf8());
74        }
75        let (..) = (self.cx2rx.push(rx), self.rx2cx.push(cx));
76        self.update_syntax(syntax, hl_state)
77    }
78
79    /// Obtain the character size, in bytes, given its position in
80    /// `self.render`. This is done in constant time by using the difference
81    /// between `self.rx2cx[rx]` and the cx for the next character.
82    pub fn get_char_size(&self, rx: usize) -> usize {
83        self.rx2cx.iter().skip(rx + 1).map(|cx| cx - self.rx2cx[rx]).find(|d| *d > 0).unwrap_or(1)
84    }
85
86    /// Update the syntax highlighting types of the row.
87    fn update_syntax(&mut self, syntax: &SyntaxConf, mut hl_state: HlState) -> HlState {
88        self.hl.clear();
89        let line = self.render.as_bytes();
90
91        // Delimiters for multi-line comments and multi-line strings, as Option<&String,
92        // &String>
93        let ml_comment_delims = syntax.ml_comment_delims.as_ref().map(|(start, end)| (start, end));
94        let ml_string_delims = syntax.ml_string_delim.as_ref().map(|x| (x, x));
95
96        'syntax_loop: while self.hl.len() < line.len() {
97            let i = self.hl.len();
98            let find_str = |s: &str| line.get(i..(i + s.len())).is_some_and(|r| r.eq(s.as_bytes()));
99
100            if hl_state == HlState::Normal && syntax.sl_comment_start.iter().any(|s| find_str(s)) {
101                self.hl.extend(repeat_n(HlType::Comment, line.len() - i));
102                continue;
103            }
104
105            // Multi-line strings and multi-line comments have the same behavior; the only
106            // differences are: the start/end delimiters, the `HLState`, the `HLType`.
107            for (delims, mstate, mtype) in &[
108                (ml_comment_delims, HlState::MultiLineComment, HlType::MlComment),
109                (ml_string_delims, HlState::MultiLineString, HlType::MlString),
110            ] {
111                if let Some((start, end)) = delims {
112                    if hl_state == *mstate {
113                        if find_str(end) {
114                            // Highlight the remaining symbols of the multi line comment end
115                            self.hl.extend(repeat_n(mtype, end.len()));
116                            hl_state = HlState::Normal;
117                        } else {
118                            self.hl.push(*mtype);
119                        }
120                        continue 'syntax_loop;
121                    } else if hl_state == HlState::Normal && find_str(start) {
122                        // Highlight the remaining symbols of the multi line comment start
123                        self.hl.extend(repeat_n(mtype, start.len()));
124                        hl_state = *mstate;
125                        continue 'syntax_loop;
126                    }
127                }
128            }
129
130            let c = line[i];
131
132            // At this point, hl_state is Normal or String
133            if let HlState::String(quote) = hl_state {
134                self.hl.push(HlType::String);
135                if c == quote {
136                    hl_state = HlState::Normal;
137                } else if c == b'\\' && i != line.len() - 1 {
138                    self.hl.push(HlType::String);
139                }
140                continue;
141            } else if syntax.sl_string_quotes.contains(&(c as char)) {
142                hl_state = HlState::String(c);
143                self.hl.push(HlType::String);
144                continue;
145            }
146
147            let prev_sep = (i == 0) || is_sep(line[i - 1]);
148
149            if syntax.highlight_numbers
150                && ((c.is_ascii_digit() && prev_sep)
151                    || (i != 0 && self.hl[i - 1] == HlType::Number && !prev_sep && !is_sep(c)))
152            {
153                self.hl.push(HlType::Number);
154                continue;
155            }
156
157            if prev_sep {
158                // This filters makes sure that names such as "in_comment" are not partially
159                // highlighted (even though "in" is a keyword in rust)
160                // The argument is the keyword that is matched at `i`.
161                let s_filter = |kw: &str| line.get(i + kw.len()).is_none_or(|c| is_sep(*c));
162                for (keyword_highlight_type, kws) in &syntax.keywords {
163                    for keyword in kws.iter().filter(|kw| find_str(kw) && s_filter(kw)) {
164                        self.hl.extend(repeat_n(*keyword_highlight_type, keyword.len()));
165                    }
166                }
167            }
168
169            self.hl.push(HlType::Normal);
170        }
171
172        // String state doesn't propagate to the next row
173        self.hl_state =
174            if matches!(hl_state, HlState::String(_)) { HlState::Normal } else { hl_state };
175        self.hl_state
176    }
177
178    /// Draw the row and write the result to a buffer. An `offset` can be given,
179    /// as well as a limit on the length of the row (`max_len`). After
180    /// writing the characters, clear the rest of the line and move the
181    /// cursor to the start of the next line.
182    pub fn draw(&self, offset: usize, max_len: usize, buffer: &mut String, use_color: bool) {
183        let mut current_hl_type = HlType::Normal;
184        let chars = self.render.chars().skip(offset).take(max_len);
185        let mut rx = self.render.chars().take(offset).map(|c| c.width().unwrap_or(1)).sum();
186        for (c, mut hl_type) in chars.zip(self.hl.iter().skip(offset)) {
187            if c.is_ascii_control() {
188                let rendered_char = if (c as u8) <= 26 { (b'@' + c as u8) as char } else { '?' };
189                push_colored(buffer, WBG, &rendered_char.to_string(), use_color);
190                // Restore previous color
191                if use_color && current_hl_type != HlType::Normal {
192                    buffer.push_str(&current_hl_type.to_string());
193                }
194            } else {
195                if let Some(match_segment) = &self.match_segment {
196                    if match_segment.contains(&rx) {
197                        // Set the highlight type to Match, i.e. set the background to cyan
198                        hl_type = &HlType::Match;
199                    } else if use_color && rx == match_segment.end {
200                        // Reset the formatting, in particular the background
201                        buffer.push_str(RESET);
202                    }
203                }
204                if use_color && current_hl_type != *hl_type {
205                    buffer.push_str(&hl_type.to_string());
206                }
207                current_hl_type = *hl_type;
208                buffer.push(c);
209            }
210            rx += c.width().unwrap_or(1);
211        }
212        buffer.push_str(if use_color { RESET } else { "" });
213    }
214}
215
216/// Return whether `c` is an ASCII separator.
217const fn is_sep(c: u8) -> bool {
218    c.is_ascii_whitespace() || c == b'\0' || (c.is_ascii_punctuation() && c != b'_')
219}
