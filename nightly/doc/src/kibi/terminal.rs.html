9/// Obtain the window size using the cursor position.
10///
11/// This function moves the cursor to the bottom-right using ANSI escape
12/// sequence `\x1b[999C\x1b[999B`, then requests the cursor position using ANSI
13/// escape sequence `\x1b[6n`. After this sequence is sent, the next characters
14/// on stdin should be `\x1b[{row};{column}R`.
15///
16/// It is used as an alternative method if `sys::get_window_size()` returns an
17/// error.
18pub fn get_window_size_using_cursor() -> Result<(usize, usize), Error> {
19    print!("{REPOSITION_CURSOR_END}{DEVICE_STATUS_REPORT}");
20    io::stdout().flush()?;
21    let mut prefix_buffer = [0u8; 2];
22    sys::stdin()?.read_exact(&mut prefix_buffer)?;
23    if prefix_buffer != [b'\x1b', b'['] {
24        return Err(Error::CursorPosition);
25    }
26    Ok((read_value_until(b';')?, read_value_until(b'R')?))
27}

29/// Read value until a certain stop byte is reached, and parse the result
30/// (pre-stop byte).
31fn read_value_until<T: std::str::FromStr>(stop_byte: u8) -> Result<T, Error> {
32    let mut buf = Vec::new();
33    sys::stdin()?.read_until(stop_byte, &mut buf)?;
34    // Check that we have reached `stop_byte`, not EOF.
35    buf.pop().filter(|u| *u == stop_byte).ok_or(Error::CursorPosition)?;
36    std::str::from_utf8(&buf).or(Err(Error::CursorPosition))?.parse().or(Err(Error::CursorPosition))
37}

39#[cfg_attr(any(windows, target_os = "wasi"), expect(clippy::trivially_copy_pass_by_ref))]
40pub fn restore_terminal(orig_term_mode: &sys::TermMode) -> io::Result<()> {
41    // Restore the original terminal mode.
42    sys::set_term_mode(orig_term_mode)?;
43    print!("{USE_MAIN_SCREEN}");
44    io::stdout().flush()
45}
