8use crate::config::{self, parse_value as pv, parse_values as pvs};

10/// Type of syntax highlighting for a single rendered character.
11///
12/// Each `HLType` is associated with a color, via its discriminant. The ANSI
13/// color is equal to the discriminant, modulo 100. The colors are described
14/// here: <https://en.wikipedia.org/wiki/ANSI_escape_code#Colors>
15#[derive(Clone, Copy, Debug, Eq, PartialEq)]
16pub enum HlType {
17    Normal = 39,     // Default foreground color
18    Number = 31,     // Red
19    Match = 46,      // Cyan
20    String = 32,     // Green
21    MlString = 132,  // Green
22    Comment = 34,    // Blue
23    MlComment = 134, // Blue
24    Keyword1 = 33,   // Yellow
25    Keyword2 = 35,   // Magenta
26}
27
28impl Display for HlType {
29    /// Write the ANSI color escape sequence for the `HLType` using the given
30    /// formatter.
31    fn fmt(&self, f: &mut Formatter) -> fmt::Result { write!(f, "\x1b[{}m", (*self as u32) % 100) }
32}
33
34/// Configuration for syntax highlighting.
35#[derive(Clone, Debug, Default, Eq, PartialEq)]
36pub struct Conf {
37    /// The name of the language, e.g. "Rust".
38    pub name: String,
39    /// Whether to highlight numbers.
40    pub highlight_numbers: bool,
41    /// Quotes for single-line strings.
42    pub sl_string_quotes: Vec<char>,
43    /// The tokens that starts a single-line comment, e.g. "//".
44    pub sl_comment_start: Vec<String>,
45    /// The tokens that start and end a multi-line comment, e.g. ("/*", "*/").
46    pub ml_comment_delims: Option<(String, String)>,
47    /// The token that start and end a multi-line strings, e.g. "\"\"\"" for
48    /// Python.
49    pub ml_string_delim: Option<String>,
50    /// Keywords to highlight and there corresponding `HLType` (typically
51    /// `HLType::Keyword1` or `HLType::Keyword2`)
52    pub keywords: Vec<(HlType, Vec<String>)>,
53}
54
55impl Conf {
56    /// Return the syntax configuration corresponding to the given file
57    /// name, if a matching INI file is found in a config directory.
58    /// If no matching configuration is found, return the default.
59    pub fn find(name: &str, data_dirs: &[String]) -> Self {
60        for data_dir in data_dirs {
61            match PathBuf::from(data_dir).join("syntax.d").read_dir() {
62                Ok(dir_entries) =>
63                    for dir_entry in dir_entries {
64                        match dir_entry.map(|dir_entry| Self::parse(&dir_entry.path())) {
65                            // sfix = suffixes
66                            Ok((sc, sfix)) if sfix.iter().any(|s| name.ends_with(s)) => return sc,
67                            Ok((..)) => (),
68                            Err(e) => eprintln!("Error iterating through {data_dir}/syntax.d: {e}"),
69                        }
70                    },
71                Err(e) if e.kind() == std::io::ErrorKind::NotFound => {}
72                Err(e) => eprintln!("Error iterating through {data_dir}/syntax.d: {e}"),
73            }
74        }
75        Self::default()
76    }
77
78    /// Load and parse a `SyntaxConf` from file.
79    pub fn parse(path: &Path) -> (Self, Vec<String>) {
80        let (mut sc, mut suffixes) = (Self::default(), Vec::new());
81        config::process_ini_file(path, &mut |key, val| {
82            match key {
83                "name" => sc.name = pv(val)?,
84                "extensions" => suffixes.extend(val.split(',').map(|u| format!(".{}", u.trim()))),
85                "highlight_numbers" => sc.highlight_numbers = pv(val)?,
86                "singleline_string_quotes" => sc.sl_string_quotes = pvs(val)?,
87                "singleline_comment_start" => sc.sl_comment_start = pvs(val)?,
88                "multiline_comment_delims" =>
89                    sc.ml_comment_delims = match val.split_once(',') {
90                        Some((v1, v2)) if !v2.contains(',') => Some((pv(v1)?, pv(v2)?)),
91                        _ => return Err(format!("Expected 2 delimiters, got {val}")),
92                    },
93                "multiline_string_delim" => sc.ml_string_delim = Some(pv(val)?),
94                "keywords_1" => sc.keywords.push((HlType::Keyword1, pvs(val)?)),
95                "keywords_2" => sc.keywords.push((HlType::Keyword2, pvs(val)?)),
96                _ => return Err(String::from("Invalid key")),
97            }
98            Ok(())
99        });
100        (sc, suffixes)
101    }
102}

104#[cfg(test)]
105#[cfg(not(target_family = "wasm"))] // No filesystem on wasm
106mod tests {
107    use std::collections::HashSet;
108    use std::fs;
109
110    use tempfile::TempDir;
111
112    use super::*;
113
114    #[test]
115    fn syntax_d_files() {
116        let mut file_count = 0;
117        let mut syntax_names = HashSet::new();
118        for path in fs::read_dir("./syntax.d").unwrap() {
119            let (conf, extensions) = Conf::parse(&path.unwrap().path());
120            assert!(!extensions.is_empty());
121            syntax_names.insert(conf.name);
122            file_count += 1;
123        }
124        assert!(file_count > 0);
125        assert_eq!(file_count, syntax_names.len());
126    }
127
128    #[test]
129    fn conf_from_invalid_path() {
130        let tmp_dir = TempDir::new().expect("Could not create temporary directory");
131        let tmp_path = tmp_dir.path().join("path_does_not_exist.ini");
132        assert_eq!(Conf::parse(&tmp_path), (Conf::default(), Vec::<String>::new()));
133    }
134}
