15#[derive(Parser, Debug)]

17pub struct Args {

22#[derive(Subcommand, Debug)]
23#[command(args_conflicts_with_subcommands = true)]
24pub enum Commands {
25    Read {
26        #[arg(short, long, value_parser)]
27        id: String,
28    },
29    /// Search various aspects within the service.
30    Write {
31        #[arg(short, long, value_parser)]
32        id: String,
33    },
34}

36// sec-test read --id 0x274b9675a25837a7446a419c68df8fc7
37//
38fn main() {
39    let args = Args::parse();
40    let mut builder = default_builder();
41    builder.filter_level(LevelFilter::Trace);
42    builder.init();
43
44    match args.command {
45        Some(command) => match command {
46            Commands::Read { id } => {
47                let id = id.trim_start_matches("0x");
48
49                let sec_ctx_id = u128::from_str_radix(id, 16).unwrap().into();
50
51                let id: u128 = 0x1000000000000a;
52
53                let map =
54                    Object::<SecCtxMap>::map(sec_ctx_id, MapFlags::READ | MapFlags::WRITE).unwrap();
55
56                println!("Object Id: {:#?}", map.id());
57
58                let res = SecCtxMap::lookup(&map, id.into());
59                println!("lookup results {:#?}", res);
60            }
61
62            Commands::Write { id } => {
63                // how to build a persistent object
64                let id: u128 = id.parse().expect("id should be valid u128");
65                let vobj = ObjectBuilder::<SecCtxMap>::default()
66                    // .persist()
67                    .build(SecCtxMap::default())
68                    .unwrap();
69
70                println!("SecCtxObjId: {}", vobj.id());
71
72                let vobj_id = vobj.id();
73
74                let cap_ptr = SecCtxMap::insert(&vobj, id.into(), CtxMapItemType::Cap);
75
76                println!("Ptr: {:#?}", cap_ptr);
77                println!("SecCtxObjId: {}", vobj_id);
78
79                let res = SecCtxMap::lookup(&vobj, id.into());
80                println!("lookup results {:#?}", res);
81
82                println!("\n\n\n============================\n\n\n");
83
84                let map =
85                    Object::<SecCtxMap>::map(vobj_id, MapFlags::READ | MapFlags::WRITE).unwrap();
86
87                println!("Object Id: {:#?}", map.id());
88
89                let res = SecCtxMap::lookup(&map, id.into());
90                println!("lookup results {:#?}", res);
91            }
92        },
93
94        None => {
95            let sec_ctx = SecCtx::default();
96
97            let target = 0x123.into();
98            let accessor = 0x321.into();
99            let prots = Protections::all();
100            let target_priv_key =
101                SigningKey::from_slice(&rand_32(), Default::default()).expect("should work");
102
103            let cap = Cap::new(
104                target,
105                accessor,
106                prots,
107                target_priv_key,
108                Default::default(),
109                Default::default(),
110                Default::default(),
111                Default::default(),
112            )
113            .unwrap();
114
115            sec_ctx.add_cap(cap);
116
117            println!("{}", sec_ctx);
118
119            let id = sec_ctx.id();
120            drop(sec_ctx);
121
122            let sec_ctx: SecCtx = id.try_into().unwrap();
123
124            println!("just read: {}", sec_ctx)
125        }
126    }
127}
