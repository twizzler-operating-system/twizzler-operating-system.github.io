87    fn wipe_old_khf_file(fs: &MutexGuard<'_, fatfs::FileSystem<D>>) {
88        let old_file = fs.root_dir().open_file("old/khf");
89        let mut old_file = match old_file {
90            Err(fatfs::Error::NotFound) => return,
91            v => v.unwrap(),
92        };
93        // override old file with zeroes
94        let extents_ct = old_file.extents().collect::<Vec<_>>().len();
95        for _ in 0..extents_ct {
96            old_file.write(&[0u8; PAGE_SIZE]).unwrap();
97        }
98        // delete old file
99        fs.root_dir().remove("old/khf").unwrap();
100    }
101    fn restore_khf(fs: &MutexGuard<'_, fatfs::FileSystem<D>>) {
102        let lethe = fs.root_dir().create_dir("lethe/").unwrap();
103        let tmp_khf = fs.root_dir().open_file("tmp/khf");
104        let old_khf = fs.root_dir().open_file("old/khf");
105        // Step one: save khf to old/khf if khf exists.
106        let step_one = || {
107            let res = lethe.rename("khf", &fs.root_dir(), "old/khf");
108            match res {
109                Err(fatfs::Error::NotFound) => {
110                    // it's fine if there currently isn't a khf,
111                    // since we're about to add one from tmp/khf.
112                    // However if there was one we should make sure to
113                    // save it.
114                }
115                r => r.unwrap(),
116            };
117        };
118        // Step two: write what's in tmp/khf to lethe/khf
119        // and delete the old khf file.
120        let step_two = || {
121            fs.root_dir().rename("tmp/khf", &lethe, "khf").unwrap();
122            Self::wipe_old_khf_file(&fs);
123        };
124        match (tmp_khf, old_khf) {
125            (Ok(_new), Ok(_old)) => {
126                // don't need to do step one since the prev khf is already
127                // in old/khf.
128                step_two();
129            }
130            (Err(fatfs::Error::NotFound), Ok(_old)) => {
131                // if there isn't a new khf and there isn't an existing
132                // khf, move the old khf to the existing khf.
133                match fs.root_dir().rename("old/khf", &lethe, "khf") {
134                    // Otherwise just delete the old khf.
135                    Err(fatfs::Error::AlreadyExists) => {
136                        // just didn't get to deleting old/khf
137                        // delete it now:
138                        Self::wipe_old_khf_file(&fs);
139                    }
140                    v => v.unwrap(),
141                };
142            }
143            (Ok(_new), Err(fatfs::Error::NotFound)) => {
144                step_one();
145                step_two();
146            }
147            (Err(fatfs::Error::NotFound), Err(fatfs::Error::NotFound)) => {
148                // how it should be after an epoch.
149            }
150            (e, e2) => {
151                e.unwrap();
152                e2.unwrap();
153                panic!("unexpected error during restoration")
154            }
155        };
156    }

234    pub fn unlink_object(&self, obj_id: u128) -> Result<(), Error> {
235        let b64 = encode_obj_id(obj_id);
236        // let (khf, wal) = (kms.khf_mut(), kms.wal_mut());
237        // khf.delete(&wal, hash_obj_id(obj_id))
238        //     .map_err(Error::other)?;
239        let extents = {
240            let mut fs = self.fs().lock().unwrap();
241            let subdir = get_dir_path(&mut fs, &b64)?;
242            let mut file = subdir.open_file(&b64)?;
243            file.extents().collect::<Vec<_>>().into_iter()
244        };
245        for extent in extents {
246            let id = disk_offset_to_id(extent?.offset);
247            let kms = self.kms();
248
249            kms.khf_lock()
250                .delete(&kms.wal_lock(), id)
251                .map_err(Error::other)?;
252        }
253        let mut fs = self.fs().lock().unwrap();
254        let subdir = get_dir_path(&mut fs, &b64)?;
255        subdir.remove(&b64)?;
256        Ok(())
257    }

292    pub fn read_exact(&self, obj_id: u128, buf: &mut [u8], off: u64) -> Result<(), Error> {
293        let b64 = encode_obj_id(obj_id);
294        let mut fs = self.fs().lock().unwrap();
295        let subdir = get_dir_path(&mut fs, &b64)?;
296        let mut file = subdir.open_file(&b64)?;
297        file.seek(fatfs::SeekFrom::Start(off))?;
298        let mut rw_proxy = ReadWriteProxy::new(
299            &mut file,
300            |disk: &mut D,
301             disk_offset: u64,
302             buffer: &mut [u8]|
303             -> Result<usize, fatfs::Error<D::Error>> {
304                let out = disk.read(buffer)?;
305                //println!("reading @ {}", disk_offset);
306                let mut cipher = self
307                    .get_symmetric_cipher(disk_offset)
308                    .map_err(Error::other)?;
309                cipher.apply_keystream(buffer);
310                Ok(out)
311            },
312            || {},
313        );
314        fatfs::Read::read_exact(&mut rw_proxy, buf)?;
315        Ok(())
316    }

332    pub fn write_all(&self, obj_id: u128, buf: &[u8], off: u64) -> Result<(), Error> {
333        let b64 = encode_obj_id(obj_id);
334        let mut fs = self.fs().lock().unwrap();
335        let subdir = get_dir_path(&mut fs, &b64)?;
336        let mut file = subdir.open_file(&b64)?;
337        let _new_pos = file.seek(fatfs::SeekFrom::Start(off))?;
338        let extents_before: HashSet<WrappedExtent> = file
339            .extents()
340            .map(|v| v.map(WrappedExtent::from))
341            .try_collect()?;
342        let mut rw_proxy = ReadWriteProxy::new(
343            &mut file,
344            || {},
345            |disk: &mut D, offset: u64, buffer: &[u8]| -> Result<usize, fatfs::Error<D::Error>> {
346                //println!("writing @ {}", offset);
347                let mut cipher = self.get_symmetric_cipher(offset)?;
348                let mut encrypted = vec![0u8; buffer.len()];
349                cipher
350                    .apply_keystream_b2b(buffer, &mut encrypted)
351                    .map_err(Error::other)?;
352                let out = disk.write(&encrypted)?;
353                Ok(out)
354            },
355        );
356        fatfs::Write::write_all(&mut rw_proxy, buf)?;
357        let extents_after: HashSet<WrappedExtent> = file
358            .extents()
359            .map(|v| v.map(WrappedExtent::from))
360            .try_collect()?;
361        // Should never add extents to a file after writing to a file.
362        assert_eq!(extents_before.difference(&extents_after).next(), None);
363        Ok(())
364    }

366    pub fn advance_epoch(&self) -> Result<(), Error> {
367        let kms = self.kms();
368        let updated_keys = kms
369            .khf_lock()
370            .update(&kms.wal_lock())
371            .map_err(Error::other)?;
372        for (id, key) in updated_keys {
373            //println!("{}", id_to_disk_offset(id));
374            let mut buf = vec![0; PAGE_SIZE];
375            let mut disk = self.fs.disk().clone();
376            let disk_offset = id_to_disk_offset(id);
377            disk.seek(SeekFrom::Start(disk_offset))?;
378            disk.read_exact(buf.as_mut_slice())?;
379            let mut cipher =
380                get_symmetric_cipher_from_key(disk_offset, key).map_err(Error::other)?;
381            cipher.apply_keystream(&mut buf);
382            disk.seek(SeekFrom::Start(disk_offset))?;
383            let mut cipher = self
384                .get_symmetric_cipher(disk_offset)
385                .map_err(Error::other)?;
386            cipher.apply_keystream(&mut buf);
387            disk.write_all(&buf)?;
388        }
389        let kms = self.kms();
390        {
391            let mut khf = kms.khf_lock();
392            let fs = self.fs().lock().unwrap();
393            fs.root_dir().create_dir("tmp/")?;
394            fs.root_dir().create_dir("old/")?;
395            khf.persist(self.root_key, "tmp/khf", &fs)
396                .map_err(Error::other)?;
397            Self::wipe_old_khf_file(&fs);
398            // let lethe = fs.root_dir().create_dir("lethe/")?;
399            Self::restore_khf(&fs);
400        }
401        kms.wal_lock().clear().map_err(Error::other)?;
402        Ok(())
403    }

412    pub fn get_lethe_state(&self) -> Result<LetheState, Error> {
413        let mut objs = Vec::new();
414        for id in self.get_all_object_ids()? {
415            let mut perobj = PerObjLetheState::default();
416            perobj.id = id;
417            let extents = self.get_obj_segments(id)?;
418            for extent in extents {
419                let chunk_id = extent.0.offset / crate::fs::PAGE_SIZE as u64;
420                let kms = self.kms();
421                let key = kms
422                    .khf_lock()
423                    .derive_mut(&kms.wal_lock(), chunk_id)
424                    .map_err(Error::other)?;
425                perobj.keys.insert(extent, key);
426            }
427            objs.push(perobj);
428        }
429        let kms = self.kms();
430        let roots = kms
431            .khf_lock()
432            .roots()
433            .iter()
434            .map(|root| (root.pos.0, root.pos.1, root.key))
435            .collect();
436        Ok(LetheState { list: objs, roots })
437    }

489fn get_symmetric_cipher_from_key(disk_offset: u64, key: [u8; 32]) -> Result<ChaCha20, Error> {
490    let chunk_id = disk_offset_to_id(disk_offset);
491    let offset = disk_offset - chunk_id;
492    let bytes = chunk_id.to_le_bytes();
493    let nonce: [u8; 12] = [
494        0, 0, 0, 0, bytes[0], bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6], bytes[7],
495    ];
496
497    let mut cipher = ChaCha20::new(&key.into(), &nonce.into());
498    cipher.seek(offset);
499    Ok(cipher)
500}

633    fn get_unique_id<OsRef: Deref<Target = LetheObjectStore<FileDisk>>>(fs: &OsRef) -> u128 {
634        let mut id: u128 = rand::random();
635        while !fs.do_create_object(id).unwrap() {
636            id = rand::random();
637        }
638        id
639    }

641    fn make_and_check_file<OsRef>(fs: &OsRef, buf1: &mut [u8], buf2: &mut [u8]) -> (Vec<u8>, u128)
642    where
643        OsRef: Deref<Target = LetheObjectStore<FileDisk>>,
644    {
645        let id: u128 = get_unique_id(fs);
646        let random_value = rand::random();
647        // println!("{}", random_value);
648        buf1.fill_with(|| random_value);
649        fs.write_all(id, buf1, 0).unwrap();
650        fs.read_exact(id, buf2, 0).unwrap();
651        assert!(buf1 == buf2);
652        (buf2.into(), id)
653    }

704    fn it_works() {
705        let mut working_bufs = (vec![0; 5000], vec![0; 5000]);
706        let mut os = OBJECT_STORE.lock().unwrap();
707        // println!("{:?}", KHF.lock().unwrap());
708        let out = (0..10)
709            .map(|_i| make_and_check_file(&os, &mut working_bufs.0, &mut working_bufs.1))
710            .collect::<Vec<_>>();
711        os.advance_epoch().unwrap();
712        os.reopen();
713
714        // println!("{:?}", KHF.lock().unwrap());
715        for (value, id) in out {
716            // make sure buf == read
717            let mut buf = vec![0; 5000];
718            let v = os.get_obj_segments(id).unwrap();
719            println!("{:?}", v);
720            os.read_exact(id, &mut buf, 0).unwrap();
721            for (i, (b1, b2)) in value.iter().zip(buf.iter()).enumerate() {
722                let diff = (*b1 as i16) - (*b2 as i16);
723                if diff != 0 {
724                    print!("D @ {i}: {diff}\t");
725                }
726            }
727            assert!(value == buf);
728            // unlink
729            os.unlink_object(id).unwrap();
730            os.advance_epoch().unwrap();
731            os.reopen();
732            // println!("{:?}", KHF.lock().unwrap());
733            // make sure object is unlinked
734            let v = os.read_exact(id, &mut buf, 0).expect_err("should be error");
735            assert!(v.kind() == std::io::ErrorKind::NotFound);
736        }
